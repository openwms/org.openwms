001package org.openwms.tms;
002
003import javax.persistence.Column;
004import javax.persistence.Embeddable;
005import javax.persistence.Temporal;
006import javax.persistence.TemporalType;
007import java.io.Serializable;
008import java.util.Date;
009
010/**
011 * A Problem is used to signal an occurred failure.
012 * 
013 * @GlossaryTerm
014 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
015 * @version $Revision$
016 * @since 0.1
017 */
018@Embeddable
019public class Problem implements Serializable {
020
021    /**
022     * Timestamp when the <code>Problem</code> occurred.
023     */
024    @Temporal(TemporalType.TIMESTAMP)
025    @Column(name = "C_PROBLEM_OCCURRED")
026    private Date occurred;
027
028    /**
029     * Message number of the <code>Problem</code>.
030     */
031    @Column(name = "C_PROBLEM_MESSAGE_NO")
032    private int messageNo;
033
034    /**
035     * Message text about the <code>Problem</code>.
036     */
037    @Column(name = "C_PROBLEM_MESSAGE")
038    private String message;
039
040    // FIXME [scherrer] : add new column with root exception
041    // private Throwable rootCause;
042
043    /* ----------------------------- methods ------------------- */
044    /**
045     * Creates a new <code>Problem</code> instance.
046     */
047    public Problem() {
048        this.occurred = new Date();
049    }
050
051    /**
052     * Create a new <code>Problem</code> instance with a message text.
053     * 
054     * @param message
055     *            text as String
056     */
057    public Problem(String message) {
058        this();
059        this.message = message;
060    }
061
062    /**
063     * Create a new <code>Problem</code> instance with a message text and a message number.
064     * 
065     * @param message
066     *            text as String
067     * @param messageNo
068     *            message number
069     */
070    public Problem(String message, int messageNo) {
071        this();
072        this.message = message;
073        this.messageNo = messageNo;
074    }
075
076    /**
077     * Return the Date when the <code>Problem</code> has occurred.
078     * 
079     * @return Date when occurred.
080     */
081    public Date getOccurred() {
082        return new Date(occurred.getTime());
083    }
084
085    /**
086     * Set the Date when the <code>Problem</code> occurred.
087     * 
088     * @param occurred
089     *            The Date to set.
090     */
091    public void setOccurred(Date occurred) {
092        this.occurred = new Date(occurred.getTime());
093    }
094
095    /**
096     * Get the messageNo.
097     * 
098     * @return The messageNo.
099     */
100    public int getMessageNo() {
101        return messageNo;
102    }
103
104    /**
105     * Set the messageNo.
106     * 
107     * @param messageNo
108     *            The messageNo to set.
109     */
110    public void setMessageNo(int messageNo) {
111        this.messageNo = messageNo;
112    }
113
114    /**
115     * Get the message.
116     * 
117     * @return The message.
118     */
119    public String getMessage() {
120        return message;
121    }
122
123    /**
124     * Set the message.
125     * 
126     * @param message
127     *            The message to set.
128     */
129    public void setMessage(String message) {
130        this.message = message;
131    }
132}

022package org.openwms.tms.targets;
023
024import java.io.Serializable;
025
026import com.fasterxml.jackson.annotation.JsonCreator;
027
028/**
029 * A Location.
030 *
031 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
032 * @version 1.0
033 * @since 1.0
034 */
035public class Location implements Target, Serializable {
036
037    private String locationId;
038    private boolean incomingActive = true;
039
040    @JsonCreator
041    public Location() {
042    }
043
044    @JsonCreator
045    public Location(String locationId) {
046        this.locationId = locationId;
047    }
048
049    public boolean isIncomingActive() {
050        return incomingActive;
051    }
052
053    public void setIncomingActive(boolean incomingActive) {
054        this.incomingActive = incomingActive;
055    }
056
057    /**
058     * Return the {@code locationId}.
059     *
060     * @return String locationId
061     */
062    @Override
063    public String toString() {
064        return locationId;
065    }
066
067    @Override
068    public String asString() {
069        return locationId;
070    }
071
072    public boolean isInfeedBlocked() {
073        return !incomingActive;
074    }
075}

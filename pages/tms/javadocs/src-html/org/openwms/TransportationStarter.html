022package org.openwms;
023
024import java.util.Locale;
025
026import com.fasterxml.jackson.annotation.JsonInclude;
027import com.fasterxml.jackson.databind.DeserializationFeature;
028import com.fasterxml.jackson.databind.ObjectMapper;
029import com.fasterxml.jackson.databind.SerializationFeature;
030import org.ameba.IDGenerator;
031import org.ameba.JdkIDGenerator;
032import org.ameba.annotation.EnableAspects;
033import org.ameba.app.SolutionApp;
034import org.ameba.http.EnableMultiTenancy;
035import org.ameba.http.RequestIDFilter;
036import org.ameba.i18n.AbstractTranslator;
037import org.ameba.i18n.Translator;
038import org.ameba.mapping.BeanMapper;
039import org.ameba.mapping.DozerMapperImpl;
040import org.openwms.tms.TMSConstants;
041import org.springframework.boot.SpringApplication;
042import org.springframework.boot.autoconfigure.SpringBootApplication;
043import org.springframework.boot.web.servlet.FilterRegistrationBean;
044import org.springframework.cloud.client.circuitbreaker.EnableCircuitBreaker;
045import org.springframework.cloud.netflix.feign.EnableFeignClients;
046import org.springframework.context.MessageSource;
047import org.springframework.context.annotation.Bean;
048import org.springframework.context.annotation.Primary;
049import org.springframework.context.annotation.aspectj.EnableSpringConfigured;
050import org.springframework.context.support.ResourceBundleMessageSource;
051import org.springframework.core.Ordered;
052import org.springframework.web.servlet.LocaleResolver;
053import org.springframework.web.servlet.i18n.LocaleChangeInterceptor;
054import org.springframework.web.servlet.i18n.SessionLocaleResolver;
055
056/**
057 * A TransportationStarter is the Spring Boot starter class of the microservice component.
058 *
059 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
060 * @since 1.0
061 */
062@EnableFeignClients
063@EnableCircuitBreaker
064@SpringBootApplication(scanBasePackageClasses = {TransportationStarter.class, SolutionApp.class})
065@EnableSpringConfigured
066@EnableAspects(propagateRootCause = true)
067@EnableMultiTenancy
068public class TransportationStarter {
069
070    /**
071     * Boot up!
072     *
073     * @param args Some args
074     */
075    public static void main(String[] args) {
076        SpringApplication.run(TransportationStarter.class, args);
077    }
078
079    public
080    @Bean
081    BeanMapper beanMapper() {
082        return new DozerMapperImpl("META-INF/dozer/tms-bean-mappings.xml");
083    }
084
085    public
086    @Primary
087    @Bean(name = TMSConstants.BEAN_NAME_OBJECTMAPPER)
088    ObjectMapper jackson2ObjectMapper() {
089        ObjectMapper om = new ObjectMapper();
090        om.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
091        om.configure(SerializationFeature.WRITE_EMPTY_JSON_ARRAYS, false);
092        om.configure(SerializationFeature.INDENT_OUTPUT, true);
093        om.configure(SerializationFeature.WRITE_NULL_MAP_VALUES, false);
094        om.setSerializationInclusion(JsonInclude.Include.NON_EMPTY);
095        return om;
096    }
097
098    /*~ ------------- i18n handling ----------- */
099    public
100    @Bean
101    LocaleResolver localeResolver() {
102        SessionLocaleResolver slr = new SessionLocaleResolver();
103        slr.setDefaultLocale(Locale.US);
104        return slr;
105    }
106
107    public
108    @Bean
109    LocaleChangeInterceptor localeChangeInterceptor() {
110        LocaleChangeInterceptor lci = new LocaleChangeInterceptor();
111        lci.setParamName("lang");
112        return lci;
113    }
114
115    public
116    @Bean
117    Translator translator() {
118        return new AbstractTranslator() {
119            @Override
120            protected MessageSource getMessageSource() {
121                return messageSource();
122            }
123        };
124    }
125
126    public
127    @Bean
128    MessageSource messageSource() {
129        ResourceBundleMessageSource nrrbm = new ResourceBundleMessageSource();
130        nrrbm.setBasename("i18n");
131        return nrrbm;
132    }
133
134    /*~ ------------- Request ID handling ----------- */
135    public
136    @Bean
137    IDGenerator<String> uuidGenerator() {
138        return new JdkIDGenerator();
139    }
140
141    public
142    @Bean
143    FilterRegistrationBean requestIDFilter(IDGenerator<String> uuidGenerator) {
144        FilterRegistrationBean frb = new FilterRegistrationBean(new RequestIDFilter(uuidGenerator));
145        frb.setOrder(Ordered.HIGHEST_PRECEDENCE + 1);
146        return frb;
147    }
148}

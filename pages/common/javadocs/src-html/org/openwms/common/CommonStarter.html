022package org.openwms.common;
023
024import org.ameba.IDGenerator;
025import org.ameba.JdkIDGenerator;
026import org.ameba.annotation.EnableAspects;
027import org.ameba.app.SolutionApp;
028import org.ameba.http.EnableMultiTenancy;
029import org.ameba.http.RequestIDFilter;
030import org.ameba.i18n.AbstractTranslator;
031import org.ameba.i18n.Translator;
032import org.ameba.mapping.BeanMapper;
033import org.ameba.mapping.DozerMapperImpl;
034import org.springframework.boot.SpringApplication;
035import org.springframework.boot.autoconfigure.SpringBootApplication;
036import org.springframework.boot.web.servlet.FilterRegistrationBean;
037import org.springframework.context.MessageSource;
038import org.springframework.context.annotation.Bean;
039import org.springframework.context.support.ResourceBundleMessageSource;
040import org.springframework.core.Ordered;
041import org.springframework.data.jpa.repository.config.EnableJpaAuditing;
042
043/**
044 * A Starter.
045 *
046 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
047 * @version 1.0
048 * @since 1.0
049 */
050@SpringBootApplication(scanBasePackageClasses = {CommonStarter.class, SolutionApp.class})
051@EnableAspects
052@EnableJpaAuditing
053@EnableMultiTenancy
054public class CommonStarter {
055
056    /**
057     * Boot up!
058     *
059     * @param args Some args
060     */
061    public static void main(String[] args) {
062        SpringApplication.run(CommonStarter.class, args);
063    }
064
065    public
066    @Bean
067    Translator translator() {
068        return new AbstractTranslator() {
069            @Override
070            protected MessageSource getMessageSource() {
071                return messageSource();
072            }
073        };
074    }
075
076    public
077    @Bean
078    MessageSource messageSource() {
079        ResourceBundleMessageSource nrrbm = new ResourceBundleMessageSource();
080        nrrbm.setBasename("i18n");
081        return nrrbm;
082    }
083
084    public
085    @Bean
086    BeanMapper beanMapper() {
087        return new DozerMapperImpl("classpath:/META-INF/dozer/common-bean-mappings.xml");
088    }
089
090    /*~ ------------- Request ID handling ----------- */
091    public
092    @Bean
093    IDGenerator<String> uuidGenerator() {
094        return new JdkIDGenerator();
095    }
096
097    public
098    @Bean
099    FilterRegistrationBean requestIDFilter(IDGenerator<String> uuidGenerator) {
100        FilterRegistrationBean frb = new FilterRegistrationBean(new RequestIDFilter(uuidGenerator));
101        frb.setOrder(Ordered.HIGHEST_PRECEDENCE + 1);
102        return frb;
103    }
104}

041/**
042 * A LocationGroup is a logical group of {@code Location}s, grouping together {@code Location}s with same characteristics.
043 *
044 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
045 * @version 1.0
046 * @GlossaryTerm
047 * @see org.openwms.common.location.Location
048 * @since 0.1

050@Entity

054    /** Unique identifier of a {@code LocationGroup}. */

211        return this.groupStateIn;

214    /**
215     * Change the infeed state of the {@code LocationGroup}.
216     *
217     * @param newGroupStateIn The state to set
218     */
219    public void setGroupStateIn(LocationGroupState newGroupStateIn) {
220        if (stateInLocker != null) {
221            throw new StateChangeException("The LocationGroup's state is blocked by any other LocationGroup and cannot be changed");
222        }
223        groupStateIn = newGroupStateIn;
224        locationGroups.forEach(lg -> lg.setGroupStateIn(newGroupStateIn, this));
225    }
226
227    /**
228     * Change the infeed state of the {@code LocationGroup}.
229     *
230     * @param newGroupStateIn The state to set
231     * @param lockLG The {@code LocationGroup} that wants to lock/unlock this {@code LocationGroup}.
232     */
233    private void setGroupStateIn(LocationGroupState newGroupStateIn, LocationGroup lockLG) {
234        if (groupStateIn == LocationGroupState.NOT_AVAILABLE && newGroupStateIn == LocationGroupState.AVAILABLE) {
235
236            // unlock
237            stateInLocker = null;
238        }
239        if (groupStateIn == LocationGroupState.AVAILABLE && newGroupStateIn == LocationGroupState.NOT_AVAILABLE) {
240
241            // lock
242            stateInLocker = lockLG;
243        }
244        groupStateIn = newGroupStateIn;
245        locationGroups.forEach(lg -> lg.setGroupStateIn(newGroupStateIn, lockLG));
246    }
247
248    /**
249     * Return the outfeed state of the {@code LocationGroup}.
250     *
251     * @return The state of outfeed
252     */
253    public LocationGroupState getGroupStateOut() {
254        return groupStateOut;
255    }
256
257    /**
258     * Set the outfeed state of the {@code LocationGroup}.
259     *
260     * @param gStateOut The state to set
261     * @param lockLg The {@code LocationGroup} that wants to lock/unlock this {@code LocationGroup}.
262     */
263    public void setGroupStateOut(LocationGroupState gStateOut, LocationGroup lockLg) {
264        if (this.groupStateOut == LocationGroupState.NOT_AVAILABLE && gStateOut == LocationGroupState.AVAILABLE
265                && (this.stateOutLocker == null || this.stateOutLocker.equals(lockLg))) {
266            this.groupStateOut = gStateOut;
267            this.stateOutLocker = null;
268            for (LocationGroup child : locationGroups) {
269                child.setGroupStateOut(gStateOut, lockLg);
270            }
271        }
272        if (this.groupStateOut == LocationGroupState.AVAILABLE && gStateOut == LocationGroupState.NOT_AVAILABLE
273                && (this.stateOutLocker == null || this.stateOutLocker.equals(lockLg))) {
274            this.groupStateOut = gStateOut;
275            this.stateOutLocker = lockLg;
276            for (LocationGroup child : locationGroups) {
277                child.setGroupStateOut(gStateOut, lockLg);
278            }
279        }
280    }

375    /**
376     * Add a {@code LocationGroup} to the list of children.
377     *
378     * @param locationGroup The {@code LocationGroup} to be added as a child
379     * @return {@literal true} if the {@code LocationGroup} was new in the collection of {@code LocationGroup}s, otherwise {@literal false}
380     */
381    public boolean addLocationGroup(LocationGroup locationGroup) {
382        if (locationGroup == null) {
383            throw new IllegalArgumentException("LocationGroup to be added is null");
384        }
385        if (locationGroup.parent != null) {
386            locationGroup.parent.removeLocationGroup(locationGroup);
387        }
388        locationGroup.parent = this;
389        locationGroup.setGroupStateIn(groupStateIn, this);
390        locationGroup.setGroupStateOut(groupStateOut, this);
391        return locationGroups.add(locationGroup);
392    }
393
394    /**
395     * Remove a {@code LocationGroup} from the list of children.
396     *
397     * @param locationGroup The {@code LocationGroup} to be removed from the list of children
398     * @return {@literal true} if the {@code LocationGroup} was found and could be removed, otherwise {@literal false}
399     */
400    public boolean removeLocationGroup(LocationGroup locationGroup) {
401        Assert.notNull(locationGroup, "LocationGroup to remove is null. this: " + this);
402        locationGroup.setParent(null);
403        return locationGroups.remove(locationGroup);
404    }

532    /**
533     * Tries to change the {@code groupStateIn} and {@code groupStateOut} of the {@code LocationGroup}. A state change is only allowed when
534     * the parent {@code LocationGroup}s state is not blocked.
535     *
536     * @param stateIn The new groupStateIn to set, or {@literal null}
537     * @param stateOut The new groupStateOut to set, or {@literal null}
538     */
539    void changeState(LocationGroupState stateIn, LocationGroupState stateOut) {
540        if (groupStateIn != stateIn && stateIn != null) {
541            // GroupStateIn changed
542            if (parent != null
543                    && parent.getGroupStateIn() == LocationGroupState.NOT_AVAILABLE
544                    && groupStateIn == LocationGroupState.AVAILABLE) {
545                throw new ServiceLayerException(
546                        "Not allowed to change GroupStateIn, parent locationGroup is not available");
547            }
548            setGroupStateIn(stateIn, this);
549        }
550        if (groupStateOut != stateOut && stateOut != null) {
551            // GroupStateOut changed
552            if (parent != null
553                    && parent.getGroupStateOut() == LocationGroupState.NOT_AVAILABLE
554                    && groupStateOut == LocationGroupState.AVAILABLE) {
555                throw new ServiceLayerException(
556                        "Not allowed to change GroupStateOut, parent locationGroup is not available");
557            }
558            setGroupStateOut(stateOut, this);
559        }
560    }

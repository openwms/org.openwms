022package org.openwms.common.units;
023
024import java.io.Serializable;
025import java.math.BigDecimal;
026
027import org.openwms.core.values.Measurable;
028import org.openwms.core.values.UnitType;
029
030/**
031 * A Weight represents a real world weight, that comes with an <code>Unit</code> and a value.
032 * 
033 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
034 * @version $Revision$
035 * @since 0.1
036 */
037public class Weight implements Measurable<BigDecimal, Weight, WeightUnit>, UnitType, Serializable {
038    private static final long serialVersionUID = -8849107834046064278L;
039
040    /** The unit of the <code>Weight</code>. */
041    private WeightUnit unit;
042    /** The magnitude of the <code>Weight</code>. */
043    private BigDecimal magnitude;
044    /** Constant for a zero value. */
045    public static final Weight ZERO = new Weight("0");
046
047    /* ----------------------------- methods ------------------- */
048    /**
049     * Accessed by persistence provider.
050     */
051    protected Weight() {
052        super();
053    }
054
055    /**
056     * Create a new <code>Weight</code>.
057     * 
058     * @param magnitude
059     *            The magnitude of the <code>Weight</code>
060     * @param unit
061     *            The unit of measure
062     */
063    public Weight(BigDecimal magnitude, WeightUnit unit) {
064        this.magnitude = magnitude;
065        this.unit = unit;
066    }
067
068    /**
069     * Create a new <code>Weight</code>.
070     * 
071     * @param magnitude
072     *            The magnitude of the <code>Weight</code> as String
073     * @param unit
074     *            The unit of measure
075     */
076    public Weight(String magnitude, WeightUnit unit) {
077        this.magnitude = new BigDecimal(magnitude);
078        this.unit = unit;
079    }
080
081    /**
082     * Create a new <code>Weight</code>.
083     * 
084     * @param magnitude
085     *            The magnitude of the <code>Weight</code>
086     */
087    public Weight(BigDecimal magnitude) {
088        this.magnitude = magnitude;
089        this.unit = WeightUnit.T.getBaseUnit();
090    }
091
092    /**
093     * Create a new <code>Weight</code>.
094     * 
095     * @param magnitude
096     *            The magnitude of the <code>Weight</code> as String
097     */
098    public Weight(String magnitude) {
099        this.magnitude = new BigDecimal(magnitude);
100        this.unit = WeightUnit.T.getBaseUnit();
101    }
102
103    /**
104     * @see org.openwms.core.values.UnitType#getMeasurable()
105     */
106    @Override
107    public Weight getMeasurable() {
108        return this;
109    }
110
111    /**
112     * Returns the magnitude of the <code>Weight</code>.
113     * 
114     * @return The magnitude
115     */
116    @Override
117    public BigDecimal getMagnitude() {
118        return magnitude;
119    }
120
121    /**
122     * Returns the unit of the <code>Weight</code>.
123     * 
124     * @return The unit
125     */
126    @Override
127    public WeightUnit getUnitType() {
128        return unit;
129    }
130
131    /**
132     * {@inheritDoc}
133     */
134    @Override
135    public boolean isZero() {
136        return this.getMagnitude().equals(BigDecimal.ZERO);
137    }
138
139    /**
140     * {@inheritDoc}
141     */
142    @Override
143    public boolean isNegative() {
144        return this.getMagnitude().signum() == -1;
145    }
146
147    /**
148     * {@inheritDoc}
149     */
150    @Override
151    public Weight convertTo(WeightUnit unt) {
152        return new Weight(getMagnitude().scaleByPowerOfTen((this.getUnitType().ordinal() - unt.ordinal()) * 3), unt);
153    }
154
155    /**
156     * {@inheritDoc}
157     */
158    @Override
159    public int compareTo(Weight o) {
160        if (o.getUnitType().ordinal() > this.getUnitType().ordinal()) {
161            return -1;
162        } else if (o.getUnitType().ordinal() < this.getUnitType().ordinal()) {
163            return 1;
164        } else {
165            return this.getMagnitude().compareTo(o.getMagnitude());
166        }
167    }
168
169    /**
170     * Use amount and unit for calculation.
171     * 
172     * @return The hashCode
173     */
174    @Override
175    public int hashCode() {
176        final int prime = 31;
177        int result = 1;
178        result = prime * result + ((getMagnitude() == null) ? 0 : getMagnitude().hashCode());
179        result = prime * result + ((getUnitType() == null) ? 0 : getUnitType().hashCode());
180        return result;
181    }
182
183    /**
184     * {@inheritDoc}
185     * 
186     * Use amount and unit for comparison.
187     */
188    @Override
189    public boolean equals(Object obj) {
190        if (this == obj) {
191            return true;
192        }
193        if (obj == null) {
194            return false;
195        }
196        if (getClass() != obj.getClass()) {
197            return false;
198        }
199        Weight other = (Weight) obj;
200        if (getMagnitude() == null) {
201            if (other.getMagnitude() != null) {
202                return false;
203            }
204        } else if (!getMagnitude().equals(other.getMagnitude())) {
205            return false;
206        }
207        if (getUnitType() != other.getUnitType()) {
208            return false;
209        }
210        return true;
211    }
212
213    /**
214     * {@inheritDoc}
215     */
216    @Override
217    public String toString() {
218        return getMagnitude() + " " + getUnitType();
219    }
220}

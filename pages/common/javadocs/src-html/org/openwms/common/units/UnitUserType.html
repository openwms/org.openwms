022package org.openwms.common.units;
023
024import java.io.Serializable;
025import java.math.BigDecimal;
026import java.sql.PreparedStatement;
027import java.sql.ResultSet;
028import java.sql.SQLException;
029
030import org.hibernate.HibernateException;
031import org.hibernate.TypeMismatchException;
032import org.hibernate.engine.spi.SessionImplementor;
033import org.hibernate.type.StandardBasicTypes;
034import org.hibernate.type.Type;
035import org.hibernate.usertype.CompositeUserType;
036import org.openwms.core.values.AbstractMeasure;
037import org.openwms.core.values.Measurable;
038import org.slf4j.Logger;
039import org.slf4j.LoggerFactory;
040
041/**
042 * An UnitUserType is used by Hibernate as converter for custom <code>Unit</code> types. Only subclasses of {@link AbstractMeasure} are
043 * supported by this type converter.
044 *
045 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
046 * @version $Revision: $
047 * @since 0.2
048 */
049public class UnitUserType implements CompositeUserType {
050
051    private static final Logger LOGGER = LoggerFactory.getLogger(UnitUserType.class);
052
053    /**
054     * {@inheritDoc}
055     * <p>
056     * We expect that every unit has two fields, named <code>unitType</code> and <code>amount</code>.
057     */
058    @Override
059    public String[] getPropertyNames() {
060        return new String[]{"unitType", "amount"};
061    }
062
063    /**
064     * {@inheritDoc}
065     * <p>
066     * We're going to persist both fields as Strings.
067     */
068    @Override
069    public Type[] getPropertyTypes() {
070        return new Type[]{StandardBasicTypes.STRING, StandardBasicTypes.STRING};
071    }
072
073    /**
074     * {@inheritDoc}
075     */
076    @Override
077    public Object getPropertyValue(Object component, int property) {
078        if (component instanceof Piece) {
079            Piece piece = (Piece) component;
080            return property == 0 ? piece.getUnitType() : piece.getMagnitude();
081        } else if (component instanceof Weight) {
082            Weight weight = (Weight) component;
083            return property == 0 ? weight.getUnitType() : weight.getMagnitude();
084        }
085        throw new TypeMismatchException("Incompatible type:" + component.getClass());
086    }
087
088    /**
089     * {@inheritDoc}
090     * <p>
091     * We have immutable types, throw an UnsupportedOperationException here.
092     */
093    @Override
094    public void setPropertyValue(Object component, int property, Object value) {
095        throw new UnsupportedOperationException("Unit types are immutable");
096    }
097
098    /**
099     * {@inheritDoc}
100     * <p>
101     * We do not know the concrete implementation here and return an Unit class type.
102     */
103    @Override
104    public Class returnedClass() {
105        return Measurable.class;
106    }
107
108    /**
109     * {@inheritDoc}
110     * <p>
111     * Delegate to Unit implementation.
112     */
113    @Override
114    public boolean equals(Object x, Object y) {
115        if (x == y) {
116            return true;
117        }
118        if (x == null || y == null) {
119            return false;
120        }
121        return x.equals(y);
122    }
123
124    /**
125     * {@inheritDoc}
126     * <p>
127     * Delegate to Unit implementation.
128     */
129    @Override
130    public int hashCode(Object x) {
131        return x.hashCode();
132    }
133
134    /**
135     * {@inheritDoc}
136     * <p>
137     * Try to re-assign the value read from the database to some type of Unit. Currently supported types: <ul> <li>Piece</li>
138     * <li>Weight</li> </ul>
139     *
140     * @throws SQLException in case of database errors
141     */
142    @Override
143    public Object nullSafeGet(ResultSet rs, String[] strings, SessionImplementor sessionImplementor, Object o) throws HibernateException, SQLException {
144        String rs0 = rs.getString(strings[0]);
145        if (rs.wasNull()) {
146            return null;
147        }
148        String[] val = rs0.split("@");
149        String unitType = val[0];
150        String unitTypeClass = val[1];
151        if (Piece.class.getCanonicalName().equals(unitTypeClass)) {
152            int amount = rs.getInt(strings[1]);
153            return new Piece(amount, PieceUnit.valueOf(unitType));
154        } else if (Weight.class.getCanonicalName().equals(unitTypeClass)) {
155            BigDecimal amount = rs.getBigDecimal(strings[1]);
156            return new Weight(amount, WeightUnit.valueOf(unitType));
157        }
158        throw new TypeMismatchException("Incompatible type: " + unitTypeClass);
159    }
160
161    /**
162     * {@inheritDoc}
163     * <p>
164     * We've to store the concrete classname as well.
165     *
166     * @throws SQLException in case of database errors
167     */
168    @Override
169    public void nullSafeSet(PreparedStatement st, Object value, int index, SessionImplementor sessionImplementor) throws HibernateException, SQLException {
170        if (value == null) {
171            st.setNull(index, StandardBasicTypes.STRING.sqlType());
172            st.setNull(index + 1, StandardBasicTypes.STRING.sqlType());
173        } else {
174            if (value instanceof Piece) {
175                Piece piece = (Piece) value;
176                st.setString(index, piece.getUnitType().toString() + "@" + Piece.class.getCanonicalName());
177                st.setString(index + 1, piece.getMagnitude().toPlainString());
178                if (LOGGER.isTraceEnabled()) {
179                    LOGGER.trace("Binding '" + piece.getUnitType().toString() + "@" + Piece.class.getCanonicalName()
180                            + "' to parameter: " + index);
181                    LOGGER.trace("Binding '" + piece.getMagnitude().toPlainString() + "' to parameter: " + (index + 1));
182                }
183            } else if (value instanceof Weight) {
184                Weight weight = (Weight) value;
185                st.setString(index, weight.getUnitType().toString() + "@" + Weight.class.getCanonicalName());
186                st.setString(index + 1, weight.getMagnitude().toPlainString());
187                if (LOGGER.isTraceEnabled()) {
188                    LOGGER.trace("Binding '" + weight.getUnitType().toString() + "@" + Weight.class.getCanonicalName()
189                            + "' to parameter: " + index);
190                    LOGGER.trace("Binding '" + weight.getMagnitude().toPlainString() + "' to parameter: " + index + 1);
191                }
192            } else {
193                throw new TypeMismatchException("Incompatible type: " + value.getClass().getCanonicalName());
194            }
195        }
196    }
197
198    /**
199     * {@inheritDoc}
200     * <p>
201     * No deep copy -> Immutable types.
202     */
203    @Override
204    public Object deepCopy(Object value) {
205        return value;
206    }
207
208    /**
209     * {@inheritDoc}
210     * <p>
211     * All Unit types aren't mutable.
212     */
213    @Override
214    public boolean isMutable() {
215        return false;
216    }
217
218    /**
219     * {@inheritDoc}
220     */
221    @Override
222    public Serializable disassemble(Object value, SessionImplementor sessionImplementor) throws HibernateException {
223        return (Serializable) value;
224    }
225
226    /**
227     * {@inheritDoc}
228     */
229    @Override
230    public Object assemble(Serializable cached, SessionImplementor sessionImplementor, Object o) throws HibernateException {
231        return cached;
232    }
233
234    /**
235     * {@inheritDoc}
236     */
237    @Override
238    public Object replace(Object original, Object o1, SessionImplementor sessionImplementor, Object o2) throws HibernateException {
239        return original;
240    }
241}

022package org.openwms.common.units;
023
024import java.io.Serializable;
025import java.math.BigDecimal;
026import java.math.RoundingMode;
027
028import org.openwms.core.values.Measurable;
029import org.openwms.core.values.UnitType;
030
031/**
032 * A Piece.
033 * 
034 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
035 * @version $Revision: $
036 * @since 0.2
037 */
038public class Piece implements Measurable<BigDecimal, Piece, PieceUnit>, UnitType, Serializable {
039
040    private static final long serialVersionUID = 5268725227649308401L;
041    private static final BigDecimal SHIFTER = new BigDecimal(12);
042
043    /** The unit of the <code>Piece</code>. */
044    private PieceUnit unitType;
045    /** The magnitude of the <code>Piece</code>. */
046    private BigDecimal magnitude;
047    /** Constant for a zero value. */
048    public static final Piece ZERO = new Piece(0);
049
050    /* ----------------------------- methods ------------------- */
051    /**
052     * Accessed by persistence provider.
053     */
054    protected Piece() {
055        super();
056    }
057
058    /**
059     * @see org.openwms.core.values.UnitType#getMeasurable()
060     */
061    @Override
062    public Piece getMeasurable() {
063        return this;
064    }
065
066    /**
067     * Create a new <code>Piece</code>.
068     * 
069     * @param magnitude
070     *            The magnitude of the <code>Piece</code>
071     * @param unitType
072     *            The unit of measure
073     */
074    public Piece(int magnitude, PieceUnit unitType) {
075        this.magnitude = new BigDecimal(magnitude);
076        this.unitType = unitType;
077    }
078
079    /**
080     * Create a new <code>Piece</code>.
081     * 
082     * @param magnitude
083     *            The magnitude of the <code>Piece</code> as int
084     */
085    public Piece(int magnitude) {
086        this.magnitude = new BigDecimal(magnitude);
087        this.unitType = PieceUnit.PC.getBaseUnit();
088    }
089
090    /**
091     * Create a new <code>Piece</code>.
092     * 
093     * @param magnitude
094     *            The magnitude of the <code>Piece</code>
095     * @param unitType
096     *            The unit of measure
097     */
098    public Piece(BigDecimal magnitude, PieceUnit unitType) {
099        this.magnitude = magnitude;
100        this.unitType = unitType;
101    }
102
103    /**
104     * Create a new <code>Piece</code>.
105     * 
106     * @param magnitude
107     *            The magnitude of the <code>Piece</code> as BigDecimal
108     */
109    public Piece(BigDecimal magnitude) {
110        this.magnitude = magnitude;
111        this.unitType = PieceUnit.PC.getBaseUnit();
112    }
113
114    /**
115     * Returns the magnitude of the <code>Piece</code>.
116     * 
117     * @return The magnitude
118     */
119    @Override
120    public BigDecimal getMagnitude() {
121        return magnitude;
122    }
123
124    /**
125     * Returns the unit of the <code>Piece</code>.
126     * 
127     * @return The unit
128     */
129    @Override
130    public PieceUnit getUnitType() {
131        return unitType;
132    }
133
134    /**
135     * {@inheritDoc}
136     */
137    @Override
138    public boolean isZero() {
139        return Piece.ZERO.equals(new Piece(this.getMagnitude(), PieceUnit.DOZ));
140    }
141
142    /**
143     * {@inheritDoc}
144     */
145    @Override
146    public boolean isNegative() {
147        return this.getMagnitude().signum() == -1;
148    }
149
150    /**
151     * {@inheritDoc}
152     */
153    @Override
154    public Piece convertTo(PieceUnit unt) {
155        if (PieceUnit.PC == unt && this.getUnitType() == PieceUnit.DOZ) {
156            return new Piece(this.getMagnitude().multiply(SHIFTER), PieceUnit.PC);
157        } else if (PieceUnit.DOZ == unt && this.getUnitType() == PieceUnit.PC) {
158            return new Piece(this.getMagnitude().divide(SHIFTER, 0, RoundingMode.DOWN), PieceUnit.DOZ);
159        }
160        return this;
161    }
162
163    /**
164     * {@inheritDoc}
165     */
166    @Override
167    public int compareTo(Piece o) {
168        if (null == o) {
169            return -1;
170        }
171        if (o.getUnitType().ordinal() > this.getUnitType().ordinal()) {
172            return compare(this.getMagnitude(), o.getMagnitude().multiply(SHIFTER));
173        } else if (o.getUnitType().ordinal() < this.getUnitType().ordinal()) {
174            return compare(this.getMagnitude().multiply(SHIFTER), o.getMagnitude());
175        }
176        return this.getMagnitude().compareTo(o.getMagnitude());
177    }
178
179    /**
180     * {@inheritDoc}
181     * 
182     * Return a combination of amount and unit, e.g. 24 PC
183     */
184    @Override
185    public String toString() {
186        return getMagnitude() + " " + getUnitType();
187    }
188
189    /**
190     * {@inheritDoc}
191     * 
192     * Uses magnitude and unitType for calculation.
193     */
194    @Override
195    public int hashCode() {
196        final int prime = 31;
197        int result = 1;
198        result = prime * result + ((magnitude == null) ? 0 : magnitude.hashCode());
199        result = prime * result + ((unitType == null) ? 0 : unitType.hashCode());
200        return result;
201    }
202
203    /**
204     * {@inheritDoc}
205     * 
206     * Uses magnitude and unitType for comparison.
207     */
208    @Override
209    public boolean equals(Object obj) {
210        if (this == obj) {
211            return true;
212        }
213        if (obj == null) {
214            return false;
215        }
216        if (getClass() != obj.getClass()) {
217            return false;
218        }
219        Piece other = (Piece) obj;
220        if (magnitude == null) {
221            if (other.magnitude != null) {
222                return false;
223            }
224        }
225        return this.compareTo(other) == 0 ? true : false;
226
227    }
228
229    private int compare(BigDecimal val1, BigDecimal val2) {
230        if (val1 == val2) {
231            return 0;
232        }
233        return val1.compareTo(val2);
234    }
235}

022package org.openwms.common.comm.api;
023
024import java.io.Serializable;
025import java.util.Date;
026
027/**
028 * A CommonMessage is the abstract superclass of all messages sent to subsystems like PLC or ERP. A CommonMessage has always a message
029 * header and a body.
030 * 
031 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
032 * @version $Revision: $
033 * @since 0.2
034 */
035public abstract class CommonMessage implements Serializable {
036
037    private static final long serialVersionUID = 1L;
038    private final CommonHeader header;
039    private String errorCode;
040    private Date created;
041
042    private static final short ERROR_CODE_LENGTH = 8;
043    private static final short DATE_LENGTH = 14;
044    private static final int MESSAGE_IDENTIFIER_LENGTH = 4;
045
046    /**
047     * Create a new CommonMessage.
048     * 
049     * @param header
050     *            The message header
051     */
052    public CommonMessage(CommonHeader header) {
053        this.header = header;
054    }
055
056    /**
057     * Return the length of a date field used in telegram messages.
058     * 
059     * @return Length of a date field
060     */
061    public static final short getDateLength() {
062        return DATE_LENGTH;
063    }
064
065    /**
066     * Return the length of an errorCode field used in telegram messages.
067     * 
068     * @return Length of an errorCode
069     */
070    public static final short getErrorCodeLength() {
071        return ERROR_CODE_LENGTH;
072    }
073
074    /**
075     * Subclasses have to return an unique, case-sensitive message identifier.
076     * 
077     * @return The message TYPE field (see OSIP specification)
078     */
079    public abstract String getMessageIdentifier();
080
081    /**
082     * Return the length of the message identifier in number of characters.
083     * 
084     * @return Number of Unicode code units of the message identifier
085     */
086    public static int getMessageIdentifierLength() {
087        return MESSAGE_IDENTIFIER_LENGTH;
088    }
089
090    /**
091     * Does this type of message needs to be replied to?
092     * 
093     * @return <code>true</code> no reply needed, otherwise <code>false</code>
094     */
095    public abstract boolean isWithoutReply();
096
097    /**
098     * {@inheritDoc}
099     */
100    @Override
101    public String toString() {
102        return header.toString();
103    }
104
105    /**
106     * Get the header.
107     * 
108     * @return header
109     */
110    public CommonHeader getHeader() {
111        return header;
112    }
113
114    /**
115     * Get the errorCode.
116     * 
117     * @return the errorCode.
118     */
119    public String getErrorCode() {
120        return errorCode;
121    }
122
123    /**
124     * Set the errorCode.
125     * 
126     * @param errorCode
127     *            The errorCode to set.
128     */
129    protected void setErrorCode(String errorCode) {
130        this.errorCode = errorCode;
131    }
132
133    /**
134     * Get the created.
135     * 
136     * @return the created.
137     */
138    public Date getCreated() {
139        if (created == null) {
140            created = new Date();
141        }
142        return created;
143    }
144
145    /**
146     * Set the created.
147     * 
148     * @param created
149     *            The created to set.
150     */
151    protected void setCreated(Date created) {
152        this.created = created;
153    }
154}

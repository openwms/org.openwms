022package org.openwms.common.comm.err.tcp;
023
024import java.text.ParseException;
025
026import org.openwms.common.comm.api.CommConstants;
027import org.openwms.common.comm.api.CommonHeader;
028import org.openwms.common.comm.api.CommonMessage;
029import org.openwms.common.comm.api.MessageMapper;
030import org.openwms.common.comm.err.ErrorMessage;
031import org.openwms.common.comm.exception.MessageMissmatchException;
032import org.openwms.common.comm.util.CommonMessageFactory;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035import org.springframework.stereotype.Component;
036
037/**
038 * An ErrorTelegramMapper transforms an incoming OSIP telegram string, that is expected to be an error telegram, into an ErrorMessage.
039 * 
040 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
041 * @version $Revision: $
042 * @since 0.2
043 */
044@Component
045public class ErrorTelegramMapper implements MessageMapper<ErrorMessage> {
046
047    private static final Logger LOGGER = LoggerFactory.getLogger(ErrorTelegramMapper.class);
048
049    /**
050     * {@inheritDoc}
051     */
052    @Override
053    public ErrorMessage mapTo(String telegram) {
054        int startPayload = CommonHeader.getHeaderLength() + forType().length();
055        int startCreateDate = startPayload + CommonMessage.getErrorCodeLength();
056        try {
057            return new ErrorMessage.Builder(CommonMessageFactory.createHeader(telegram))
058                    .withErrorCode(telegram.substring(startPayload, startCreateDate))
059                    .withCreateDate(
060                            CommConstants.asDate(telegram.substring(startCreateDate,
061                                    startCreateDate + CommonMessage.getDateLength()))).build();
062        } catch (ParseException e) {
063            if (LOGGER.isErrorEnabled()) {
064                LOGGER.error("Error while parsing telegram:" + e.getMessage());
065            }
066            throw new MessageMissmatchException(e.getMessage());
067        }
068    }
069
070    /**
071     * {@inheritDoc}
072     */
073    @Override
074    public String forType() {
075        return ErrorMessage.IDENTIFIER;
076    }
077}

022package org.openwms.common.comm.api;
023
024import java.io.Serializable;
025
026import org.apache.commons.lang3.StringUtils;
027
028/**
029 * A CommonHeader represents the header part of a CommonMessage.
030 * 
031 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
032 * @version $Revision: $
033 * @since 0.2
034 */
035public class CommonHeader implements Serializable {
036
037    private static final long serialVersionUID = 1L;
038    private String sync;
039    private short messageLength;
040    private String sender;
041    private String receiver;
042    private short sequenceNo;
043
044    private static final short LENGTH_SYNC_FIELD = 3;
045    private static final short LENGTH_MESSAGE_LENGTH_FIELD = 5;
046    private static final short LENGTH_SENDER_FIELD = 5;
047    private static final short LENGTH_RECEIVER_FIELD = 5;
048    private static final short LENGTH_SEQUENCE_NO_FIELD = 5;
049
050    /**
051     * Create a new CommonHeader.
052     * 
053     * @param sync
054     *            The sync field
055     * @param messageLength
056     *            The length of the message
057     * @param sender
058     *            The sender identifier
059     * @param receiver
060     *            The receiver identifier
061     * @param sequenceNo
062     *            An incremental sequence number
063     */
064    public CommonHeader(String sync, short messageLength, String sender, String receiver, short sequenceNo) {
065        super();
066        this.sync = sync;
067        this.messageLength = messageLength;
068        this.sender = sender;
069        this.receiver = receiver;
070        this.sequenceNo = sequenceNo;
071    }
072
073    /**
074     * Create a new CommonHeader.
075     *
076     * @param commonHeader The header to get the values from.
077     */
078    public CommonHeader(CommonHeader commonHeader) {
079        this(commonHeader.getSync(), commonHeader.getMessageLength(), commonHeader.getSender(), commonHeader.getReceiver(), commonHeader.getSequenceNo());
080    }
081
082    /**
083     * Create a new CommonHeader.
084     * 
085     * @param telegram
086     *            The telegram String
087     */
088    // TODO [scherrer] : remove this it is specific to telegrams.
089    public CommonHeader(String telegram) {
090        this.sync = telegram.substring(0, getLengthSyncField());
091
092        int start = sync.length();
093        int end = start + getLengthTelegramLengthField();
094        this.messageLength = Short.parseShort(telegram.substring(start, end));
095
096        start = end;
097        end += getLengthSenderField();
098        this.sender = telegram.substring(start, end);
099
100        start = end;
101        end += getLengthReceiverField();
102        this.receiver = telegram.substring(start, end);
103
104        start = end;
105        end += getLengthSequenceNoField();
106        this.sequenceNo = Short.parseShort(telegram.substring(start, end));
107    }
108
109    /**
110     * Return the number of characters the message header allocates.
111     * 
112     * @return The header length in characters
113     */
114    public static final short getHeaderLength() {
115        return LENGTH_SYNC_FIELD + LENGTH_MESSAGE_LENGTH_FIELD + LENGTH_RECEIVER_FIELD + LENGTH_SENDER_FIELD
116                + LENGTH_SEQUENCE_NO_FIELD;
117    }
118
119    /**
120     * Get the lengthSyncField.
121     * 
122     * @return the lengthSyncField.
123     */
124    public static short getLengthSyncField() {
125        return LENGTH_SYNC_FIELD;
126    }
127
128    /**
129     * Get the lengthTelegramLengthField.
130     * 
131     * @return the lengthTelegramLengthField.
132     */
133    public static short getLengthTelegramLengthField() {
134        return LENGTH_MESSAGE_LENGTH_FIELD;
135    }
136
137    /**
138     * Get the lengthReceiverField.
139     * 
140     * @return the lengthReceiverField.
141     */
142    public static short getLengthReceiverField() {
143        return LENGTH_RECEIVER_FIELD;
144    }
145
146    /**
147     * Get the lengthSenderField.
148     * 
149     * @return the lengthSenderField.
150     */
151    public static short getLengthSenderField() {
152        return LENGTH_SENDER_FIELD;
153    }
154
155    /**
156     * Get the lengthSequenceNoField.
157     * 
158     * @return the lengthSequenceNoField.
159     */
160    public static short getLengthSequenceNoField() {
161        return LENGTH_SEQUENCE_NO_FIELD;
162    }
163
164    /**
165     * Get the sync.
166     * 
167     * @return the sync.
168     */
169    public String getSync() {
170        return sync;
171    }
172
173    /**
174     * Set the sync.
175     * 
176     * @param sync
177     *            The sync to set.
178     */
179    public void setSync(String sync) {
180        this.sync = sync;
181    }
182
183    /**
184     * Get the messageLength.
185     * 
186     * @return the messageLength.
187     */
188    public short getMessageLength() {
189        return messageLength;
190    }
191
192    /**
193     * Set the messageLength.
194     * 
195     * @param messageLength
196     *            The messageLength to set.
197     */
198    public void setMessageLength(short messageLength) {
199        this.messageLength = messageLength;
200    }
201
202    /**
203     * Get the sender.
204     * 
205     * @return the sender.
206     */
207    public String getSender() {
208        return sender;
209    }
210
211    /**
212     * Set the sender.
213     * 
214     * @param sender
215     *            The sender to set.
216     */
217    public void setSender(String sender) {
218        this.sender = sender;
219    }
220
221    /**
222     * Get the receiver.
223     * 
224     * @return the receiver.
225     */
226    public String getReceiver() {
227        return receiver;
228    }
229
230    /**
231     * Set the receiver.
232     * 
233     * @param receiver
234     *            The receiver to set.
235     */
236    public void setReceiver(String receiver) {
237        this.receiver = receiver;
238    }
239
240    /**
241     * Get the sequenceNo.
242     * 
243     * @return the sequenceNo.
244     */
245    public short getSequenceNo() {
246        return sequenceNo;
247    }
248
249    /**
250     * Set the sequenceNo.
251     * 
252     * @param sequenceNo
253     *            The sequenceNo to set.
254     */
255    public void setSequenceNo(short sequenceNo) {
256        this.sequenceNo = sequenceNo;
257    }
258
259    /**
260     * {@inheritDoc}
261     */
262    @Override
263    public int hashCode() {
264        final int prime = 31;
265        int result = 1;
266        result = prime * result + ((receiver == null) ? 0 : receiver.hashCode());
267        result = prime * result + ((sender == null) ? 0 : sender.hashCode());
268        result = prime * result + sequenceNo;
269        result = prime * result + ((sync == null) ? 0 : sync.hashCode());
270        result = prime * result + messageLength;
271        return result;
272    }
273
274    /**
275     * {@inheritDoc}
276     */
277    @Override
278    public boolean equals(Object obj) {
279        if (this == obj) {
280            return true;
281        }
282        if (obj == null) {
283            return false;
284        }
285        if (getClass() != obj.getClass()) {
286            return false;
287        }
288        CommonHeader other = (CommonHeader) obj;
289        if (receiver == null) {
290            if (other.receiver != null) {
291                return false;
292            }
293        } else if (!receiver.equals(other.receiver)) {
294            return false;
295        }
296        if (sender == null) {
297            if (other.sender != null) {
298                return false;
299            }
300        } else if (!sender.equals(other.sender)) {
301            return false;
302        }
303        if (sequenceNo != other.sequenceNo) {
304            return false;
305        }
306        if (sync == null) {
307            if (other.sync != null) {
308                return false;
309            }
310        } else if (!sync.equals(other.sync)) {
311            return false;
312        }
313        if (messageLength != other.messageLength) {
314            return false;
315        }
316        return true;
317    }
318
319    /**
320     * {@inheritDoc}
321     */
322    @Override
323    public String toString() {
324        return new StringBuilder().append(this.sync).append(StringUtils.leftPad("" + this.messageLength, LENGTH_MESSAGE_LENGTH_FIELD, "0")).append(this.sender)
325                .append(this.receiver).append(StringUtils.leftPad("" + this.sequenceNo, LENGTH_SEQUENCE_NO_FIELD, "0")).toString();
326    }
327}

022package org.openwms.core.search;
023
024import javax.xml.bind.annotation.XmlAccessType;
025import javax.xml.bind.annotation.XmlAccessorType;
026import javax.xml.bind.annotation.XmlAttribute;
027import javax.xml.bind.annotation.XmlType;
028import java.io.Serializable;
029import java.util.ArrayList;
030import java.util.List;
031
032/**
033 * A Tag represents a search keyword in the OpenWMS.org ActionSearch concept. Each {@link Action} consists of multiple Tags.
034 *
035 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
036 * @version 0.2
037 * @GlossaryTerm
038 * @since 0.2
039 */
040@XmlAccessorType(XmlAccessType.FIELD)
041@XmlType(name = "tag")
042public class Tag implements Serializable {
043
044    /**
045     * The unique name of this Tag.
046     */
047    @XmlAttribute(required = true)
048    private String name;
049
050    /**
051     * Other possible names for this tag. Used for internationalization.
052     */
053    private List<String> aliases = new ArrayList<>();
054
055    /* ----------------------------- constructors ------------------- */
056
057    /**
058     * Accessed by JAXB.
059     */
060    public Tag() {
061        super();
062    }
063
064    /**
065     * Create a new Tag.
066     *
067     * @param name The initial name
068     */
069    protected Tag(String name) {
070        super();
071        this.name = name;
072    }
073
074    /* ----------------------------- methods ------------------- */
075
076    /**
077     * Get the aliases.
078     *
079     * @return the aliases.
080     */
081    public List<String> getAliases() {
082        return aliases;
083    }
084
085    /**
086     * Set the aliases.
087     *
088     * @param aliases The aliases to set.
089     */
090    public void setAliases(List<String> aliases) {
091        this.aliases = aliases;
092    }
093
094    /**
095     * Get the name.
096     *
097     * @return the name.
098     */
099    public String getName() {
100        return name;
101    }
102
103    /**
104     * Set the name.
105     *
106     * @param name The name to set.
107     */
108    public void setName(String name) {
109        this.name = name;
110    }
111
112    /**
113     * Find the given String in the name of the Tag.
114     *
115     * @param word The String to search for
116     * @return 0 if {@code name} is not part of the tag name, otherwise some positive value
117     */
118    public int matches(String word) {
119        return name.toLowerCase().indexOf(word.toLowerCase()) + 1;
120    }
121}

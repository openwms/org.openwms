022package org.openwms.core.search;
023
024import javax.xml.bind.annotation.XmlAccessType;
025import javax.xml.bind.annotation.XmlAccessorType;
026import javax.xml.bind.annotation.XmlAttribute;
027import javax.xml.bind.annotation.XmlElement;
028import javax.xml.bind.annotation.XmlElementWrapper;
029import javax.xml.bind.annotation.XmlID;
030import javax.xml.bind.annotation.XmlSchemaType;
031import javax.xml.bind.annotation.XmlTransient;
032import javax.xml.bind.annotation.XmlType;
033import javax.xml.bind.annotation.adapters.CollapsedStringAdapter;
034import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
035import java.io.Serializable;
036import java.util.ArrayList;
037import java.util.List;
038
039/**
040 * An Action represents a possible UI action an User can take. Each Action has a resulting URL to a webpage and a descriptive text that is
041 * displayed in the UI. Additionally a field {@code weight} is used to count how many times the User has chosen this Action.
042 *
043 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
044 * @version 0.2
045 * @GlossaryTerm
046 * @since 0.2
047 */
048@XmlAccessorType(XmlAccessType.FIELD)
049@XmlType(name = "action", propOrder = {"tags"})
050public class Action implements Serializable {
051
052    /**
053     * The unique name of this Action.
054     */
055    @XmlAttribute(required = true)
056    @XmlJavaTypeAdapter(CollapsedStringAdapter.class)
057    @XmlID
058    @XmlSchemaType(name = "ID")
059    private String id;
060
061    @XmlAttribute(required = true)
062    private String url;
063
064    /**
065     * Text full text of the Action that is displayed in the UI.
066     */
067    @XmlAttribute(required = true)
068    private String text;
069
070    /**
071     *
072     */
073    @XmlTransient
074    private int weight = 0;
075
076    /**
077     * An Action has one ore more Tags. With Tags an Action can be found.
078     */
079    @XmlElementWrapper(name = "tags")
080    @XmlElement(name = "tag")
081    // FIXME [scherrer] : This list is not resolved by JAXB
082    private List<Tag> tags = new ArrayList<>();
083
084    /* ----------------------------- constructors ------------------- */
085
086    /**
087     * Create a new Action.
088     */
089    public Action() {
090        super();
091    }
092
093    /* ----------------------------- methods ------------------- */
094
095    /**
096     * Calculate and return a rating for a list of words. Every match between a word and one of the tags, increases the rating. In case none
097     * of the tags matches, 0 is returned.
098     *
099     * @param words The Array of words to calculate the rating
100     * @return 0 when none of the words matches, or any other positive value in case of matches
101     */
102    public int rate(String... words) {
103        int result = 0;
104        for (String word : words) {
105            for (Tag tag : tags) {
106                if (tag.matches(word) > 0) {
107                    result++;
108                    break;
109                }
110            }
111        }
112        return result;
113    }
114
115    /**
116     * Get the name.
117     *
118     * @return the name.
119     */
120    public String getName() {
121        return id;
122    }
123
124    /**
125     * Set the name.
126     *
127     * @param name The name to set.
128     */
129    public void setName(String name) {
130        this.id = name;
131    }
132
133    /**
134     * Get the name.
135     *
136     * @return the name.
137     */
138    public String getId() {
139        return id;
140    }
141
142    /**
143     * Set the name.
144     *
145     * @param id The name to set.
146     */
147    public void setId(String id) {
148        this.id = id;
149    }
150
151    /**
152     * Get the url.
153     *
154     * @return the url.
155     */
156    public String getUrl() {
157        return url;
158    }
159
160    /**
161     * Set the url.
162     *
163     * @param url The url to set.
164     */
165    public void setUrl(String url) {
166        this.url = url;
167    }
168
169    /**
170     * Get the text.
171     *
172     * @return the text.
173     */
174    public String getText() {
175        return text;
176    }
177
178    /**
179     * Set the text.
180     *
181     * @param text The text to set.
182     */
183    public void setText(String text) {
184        this.text = text;
185    }
186
187    /**
188     * Get the weight.
189     *
190     * @return the weight.
191     */
192    public int getWeight() {
193        return weight;
194    }
195
196    /**
197     * Set the weight.
198     *
199     * @param weight The weight to set.
200     */
201    public void setWeight(int weight) {
202        this.weight = weight;
203    }
204
205    /**
206     * Increase weight by 1 and return the new weight.
207     *
208     * @return The weight increased by 1
209     */
210    public int increaseWeight() {
211        return weight++;
212    }
213
214    /**
215     * Decrease weight by 1 and return the new weight.
216     *
217     * @return The weight decreased by 1
218     */
219    public int decreaseWeight() {
220        return --weight;
221    }
222
223    /**
224     * Get the tags.
225     *
226     * @return the tags.
227     */
228    public List<Tag> getTags() {
229        return tags;
230    }
231
232    /**
233     * Set the tags.
234     *
235     * @param tags The tags to set.
236     */
237    public void setTags(List<Tag> tags) {
238        this.tags = tags;
239    }
240
241    /**
242     * {@inheritDoc}
243     */
244    @Override
245    public String toString() {
246        return getName() + getText() + getUrl();
247    }
248
249    /**
250     * A Builder class for <code>Action</code>s.
251     *
252     * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
253     * @version $Revision$
254     * @since 0.2
255     */
256    public static class Builder {
257
258        private Action action;
259
260        /**
261         * Create a new Action.Builder.
262         *
263         * @param name An initial name
264         */
265        public Builder(String name) {
266            action = new Action();
267            action.setName(name);
268        }
269
270        /**
271         * Add a text to the Action.
272         *
273         * @param text The text
274         * @return The Builder
275         */
276        public Builder withText(String text) {
277            action.setText(text);
278            return this;
279        }
280
281        /**
282         * Add a Tag to the Action.
283         *
284         * @param tag The Tag
285         * @return The Builder
286         */
287        public Builder withTag(Tag tag) {
288            action.getTags().add(tag);
289            return this;
290        }
291
292        /**
293         * Build and return the Action.
294         *
295         * @return The Action
296         */
297        public Action build() {
298            return action;
299        }
300    }
301}

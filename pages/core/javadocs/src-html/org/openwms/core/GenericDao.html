022package org.openwms.core;
023
024import java.io.Serializable;
025import java.util.List;
026import java.util.Map;
027
028/**
029 * A GenericDao is a generic interface declaration that defines simple CRUD operations.
030 * <p>
031 * This interface provides basic functionality to create/read/update and delete entity classes. Entities can be simple POJO classes without
032 * any ORM dependencies. Merely implementations of this interface knows about the way the entity is been persisted.
033 * </p>
034 * 
035 * @param <T>
036 *            Any serializable type, mostly an entity class type
037 * @param <ID>
038 *            The type of the entity class' unique id
039 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
040 * @version $Revision$
041 * @since 0.1
042 */
043public interface GenericDao<T extends AbstractEntity<ID>, ID extends Serializable> {
044
045    /**
046     * Suffix for the FIND_ALL named query.
047     */
048    String FIND_ALL = ".findAll";
049
050    /**
051     * Suffix for the FIND_BY_ID named query.
052     */
053    String FIND_BY_ID = ".findById";
054
055    /**
056     * Find and return the entity identified by the technical key.
057     * 
058     * @param id
059     *            - Unique technical key of the entity
060     * @return Entity class identified by id
061     * @throws IllegalArgumentException
062     *             if <tt>id</tt> is <code>null</code>
063     */
064    T findById(ID id);
065
066    /**
067     * Find all entities and return them as a {@link java.util.List}.
068     * 
069     * @return List of all entities
070     */
071    List<T> findAll();
072
073    /**
074     * Use a named query to find all entities. Pass in the name of the <code>NamedQuery</code> and a parameter map.
075     * 
076     * @param queryName
077     *            - Defined name of the NamedQuery
078     * @param params
079     *            - Map of parameters to pass to the query
080     * @return List of found entity classes
081     */
082    List<T> findByNamedParameters(String queryName, Map<String, ?> params);
083
084    /**
085     * Use an own JPA query to fetch entities.
086     * 
087     * @param queryName
088     *            - Defined name of the NamedQuery
089     * @param values
090     *            - A list of values to use as parameters
091     * @return A list of all entities
092     */
093    List<T> findByPositionalParameters(String queryName, Object... values);
094
095    /**
096     * Find and return the Entity identified by it's natural unique id.
097     * 
098     * @param id
099     *            - Natural key to lookup the Entity (also known as business key)
100     * @return The Entity instance or <code>null</code>
101     */
102    T findByUniqueId(Serializable id);
103
104    /**
105     * Synchronize an entity with the persistence layer and return it.
106     * 
107     * @param entity
108     *            - Entity instance to be synchronized with the persistence layer
109     * @return The synchronized entity instance. If JPA is used as implementation, the returned instance is managed.
110     */
111    T save(T entity);
112
113    /**
114     * Removes an already persistent entity.
115     * 
116     * @param entity
117     *            - Entity instance to be removed
118     * @throws IllegalArgumentException
119     *             if the instance is not an entity or is a detached entity
120     */
121    void remove(T entity);
122
123    /**
124     * Persist a transient entity.
125     * 
126     * @param entity
127     *            - Entity instance to be persisted
128     */
129    void persist(T entity);
130}

022package org.openwms.core.http;
023
024import javax.servlet.http.HttpServletRequest;
025import javax.validation.ValidationException;
026import javax.validation.Validator;
027
028import org.ameba.http.AbstractBase;
029import org.ameba.http.Response;
030import org.openwms.core.exception.ExceptionCodes;
031import org.springframework.beans.factory.annotation.Autowired;
032import org.springframework.context.MessageSource;
033import org.springframework.http.HttpStatus;
034import org.springframework.http.ResponseEntity;
035import org.springframework.util.MultiValueMap;
036import org.springframework.web.bind.MethodArgumentNotValidException;
037import org.springframework.web.bind.annotation.ExceptionHandler;
038import org.springframework.web.util.UriTemplate;
039
040/**
041 * A AbstractWebController.
042 *
043 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
044 * @version $Revision: $
045 * @since 0.1
046 */
047public abstract class AbstractWebController {
048
049    @Autowired
050    private MessageSource messageSource;
051    @Autowired
052    private Validator validator;
053
054    /**
055     * All general exceptions thrown by services are caught here and translated into http conform responses with a status code {@code 500
056     * Internal Server Error}.
057     *
058     * @param ex The exception occurred
059     * @return A response object wraps the server result
060     */
061    @ExceptionHandler(Exception.class)
062    public ResponseEntity<Response> handleException(Exception ex) {
063        if (ex.getClass().equals(HttpBusinessException.class)) {
064            HttpBusinessException e = (HttpBusinessException) ex;
065            return new ResponseEntity<>(new Response(ex.getMessage(), e.getHttpStatus().toString()), e.getHttpStatus());
066        }
067        if (ex.getClass().equals(ValidationException.class)) {
068            return new ResponseEntity<>(new Response(ex.getMessage(), HttpStatus.INTERNAL_SERVER_ERROR.toString()),
069                    HttpStatus.INTERNAL_SERVER_ERROR);
070        }
071        return new ResponseEntity<>(new Response(ex.getMessage(), HttpStatus.INTERNAL_SERVER_ERROR.toString()),
072                HttpStatus.INTERNAL_SERVER_ERROR);
073    }
074
075    /**
076     * Transforming {@link MethodArgumentNotValidException} {@link ValidationException} into server error {@code 500 Internal Server Error}
077     * responses with according validation error message text.
078     *
079     * @return A response object wraps the server result
080     */
081    @ExceptionHandler({MethodArgumentNotValidException.class, ValidationException.class})
082    public ResponseEntity<Response> handleValidationException() {
083        return new ResponseEntity<>(new Response(translate(ExceptionCodes.VALIDATION_ERROR),
084                HttpStatus.INTERNAL_SERVER_ERROR.toString()), HttpStatus.INTERNAL_SERVER_ERROR);
085    }
086
087    /**
088     * Get the messageSource.
089     *
090     * @param key The error code to search message text for
091     * @param objects Any arguments that are passed into the message text
092     * @return the messageSource.
093     */
094    protected String translate(String key, Object... objects) {
095        return messageSource.getMessage(key, objects, null);
096    }
097
098    /**
099     * Build a response object that signals a not-okay response with a given
100     * status {@code code}.
101     *
102     * @param <T> Some type extending the AbstractBase entity
103     * @param code The status code to set as response code
104     * @param msg The error message passed to the caller
105     * @param msgKey The error message key passed to the caller
106     * @param params A set of Serializable objects that are passed to the caller
107     * @return A ResponseEntity with status {@code code}
108     */
109    protected <T extends AbstractBase> ResponseEntity<Response<T>> buildResponse(HttpStatus code, String msg, String msgKey, T... params) {
110        Response result = new Response(msg, msgKey, code.toString(), params);
111        //result.add(linkTo(this.getClass()).withSelfRel());
112        return new ResponseEntity<>(result, code);
113    }
114
115    /**
116     * Build a response object that signals a not-okay response with a given
117     * status {@code code} and with given http headers.
118     *
119     * @param <T> Some type extending the AbstractBase entity
120     * @param code The status code to set as response code
121     * @param msg The error message passed to the caller
122     * @param msgKey The error message key passed to the caller
123     * @param headers The map of headers.
124     * @param params A set of Serializable objects that are passed to the caller
125     * @return A ResponseEntity with status {@code code}
126     */
127    protected <T extends AbstractBase> ResponseEntity<Response<T>> buildResponse(HttpStatus code, String msg, String msgKey, MultiValueMap<String, String> headers, T... params) {
128        Response result = new Response(msg, msgKey, code.toString(), params);
129        return new ResponseEntity<>(result, headers, code);
130    }
131
132    /**
133     * Build an response object that signals a success response to the caller.
134     *
135     * @param <T> Some type extending the AbstractBase entity
136     * @param params A set of Serializable objects that are passed to the caller
137     * @return A ResponseEntity with status {@link HttpStatus#OK}
138     */
139    protected <T extends AbstractBase> ResponseEntity<Response<T>> buildOKResponse(T... params) {
140        return buildResponse(HttpStatus.OK, "", "", params);
141    }
142
143    /**
144     * Build a response object that signals a not-okay response with a given
145     * status {@code code}.
146     *
147     * @param <T> Some type extending the AbstractBase entity
148     * @param code The status code to set as response code
149     * @param msg The error message passed to the caller
150     * @param msgKey The error message key passed to the caller
151     * @param params A set of Serializable objects that are passed to the caller
152     * @return A ResponseEntity with status {@code code}
153     */
154    protected <T extends AbstractBase> ResponseEntity<Response<T>> buildNOKResponseWithKey(HttpStatus code, String msg, String msgKey, T... params) {
155        return buildResponse(code, msg, msgKey, params);
156    }
157
158    /**
159     * Build a response object that signals a not-okay response with a given
160     * status {@code code}.
161     *
162     * @param <T> Some type extending the AbstractBase entity
163     * @param code The status code to set as response code
164     * @param msg The error message passed to the caller
165     * @param params A set of Serializable objects that are passed to the caller
166     * @return A ResponseEntity with status {@code code}
167     */
168    protected <T extends AbstractBase> ResponseEntity<Response<T>> buildNOKResponse(HttpStatus code, String msg, T... params) {
169        return buildResponse(code, msg, "", params);
170    }
171
172    /**
173     * Append the ID of the object that was created to the original request URL and return it.
174     *
175     * @param req The HttpServletRequest object
176     * @param objId The ID to append
177     * @return The complete appended URL
178     */
179    protected String getLocationForCreatedResource(HttpServletRequest req, String objId) {
180        StringBuffer url = req.getRequestURL();
181        UriTemplate template = new UriTemplate(url.append("/{objId}/").toString());
182        return template.expand(objId).toASCIIString();
183    }
184
185}

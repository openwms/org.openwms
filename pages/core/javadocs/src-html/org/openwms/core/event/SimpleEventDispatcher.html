022package org.openwms.core.event;
023
024import java.util.HashMap;
025import java.util.HashSet;
026import java.util.Map;
027import java.util.Set;
028
029import org.ameba.exception.ServiceLayerException;
030import org.springframework.beans.factory.annotation.Autowired;
031import org.springframework.context.ApplicationContext;
032import org.springframework.stereotype.Component;
033
034/**
035 * A SimpleEventDispatcher.
036 *
037 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
038 * @version $Revision: $
039 * @since 0.2
040 */
041@Component(value = SimpleEventDispatcher.COMPONENT_NAME)
042public class SimpleEventDispatcher implements EventDispatcher {
043
044    private final Map<Class<? extends RootApplicationEvent>, Set<EventListener>> subscriptions = new HashMap<>();
045    /** Springs service name. */
046    public static final String COMPONENT_NAME = "simpleEventDispatcher";
047    @Autowired
048    private ApplicationContext ctx;
049
050    /**
051     * @see org.openwms.core.event.EventBroker#subscribe(java.lang.Class, org.openwms.core.event.EventListener)
052     */
053    @Override
054    public void subscribe(Class<? extends RootApplicationEvent> event, EventListener listener) {
055        Set<EventListener> listeners;
056        synchronized (subscriptions) {
057            if (subscriptions.containsKey(event)) {
058                listeners = subscriptions.get(event);
059                if (null == listeners) {
060                    listeners = new HashSet<>(1);
061                    subscriptions.put(event, listeners);
062                }
063                listeners.add(listener);
064            } else {
065                listeners = new HashSet<>(1);
066                listeners.add(listener);
067                subscriptions.put(event, listeners);
068            }
069        }
070    }
071
072    /**
073     * @see org.openwms.core.event.EventBroker#subscribe(java.lang.Class, java.lang.String)
074     */
075    @Override
076    public void subscribe(Class<? extends RootApplicationEvent> event, String listenerBeanName) {
077        Object instance = ctx.getBean(listenerBeanName);
078        if (instance instanceof EventListener) {
079            subscribe(event, (EventListener) instance);
080        } else {
081            throw new ServiceLayerException("The bean with name " + listenerBeanName
082                    + " is not of type EventListener and cannot subscribe to events");
083        }
084    }
085
086    /**
087     * @see org.openwms.core.event.EventBroker#unsubscribe(java.lang.Class, org.openwms.core.event.EventListener)
088     */
089    @Override
090    public void unsubscribe(Class<? extends RootApplicationEvent> event, EventListener listener) {
091        if (subscriptions.containsKey(event)) {
092            synchronized (subscriptions.get(event)) {
093                subscriptions.get(event).remove(listener);
094            }
095        }
096    }
097
098    /**
099     * @see org.openwms.core.event.EventBroker#unsubscribe(java.lang.Class, java.lang.String)
100     */
101    @Override
102    public void unsubscribe(Class<? extends RootApplicationEvent> event, String listenerBeanName) {
103        Object instance = ctx.getBean(listenerBeanName);
104        if (instance instanceof EventListener) {
105            unsubscribe(event, (EventListener) instance);
106        } else {
107            throw new ServiceLayerException("The bean with name " + listenerBeanName
108                    + " is not of type EventListener and cannot unsubscribe from events");
109        }
110    }
111
112    /**
113     * @see org.openwms.core.service.spring.event.EventDispatcher#dispatch(org.openwms.core.event.RootApplicationEvent)
114     */
115    @Override
116    public <T extends RootApplicationEvent> void dispatch(T event) {
117        if (null == event || !subscriptions.containsKey(event.getClass())) {
118            return;
119        }
120        synchronized (subscriptions.get(event.getClass())) {
121            Set<EventListener> listeners = subscriptions.get(event.getClass());
122            for (EventListener eventListener : listeners) {
123                eventListener.onEvent(event);
124            }
125        }
126    }
127}

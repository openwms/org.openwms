022package org.openwms.core.listener;
023
024import org.ameba.integration.jpa.BaseEntity;
025import org.openwms.core.AbstractEntity;
026
027/**
028 * An OnRemovalListener is able to hook in the lifecycle of an entity class. A class implementing this interface is called before an entity
029 * of type <code>T</code> is removed.
030 *
031 * @param <T> Any kind of {@link AbstractEntity}
032 * @author <a href="mailto:russelltina@users.sourceforge.net">Tina Russell</a>
033 * @version $Revision$
034 * @since 0.1
035 */
036public interface OnRemovalListener<T extends BaseEntity> {
037
038    /**
039     * Do something prior the <code>entity</code> instance is been removed.
040     *
041     * @param entity The instance to be removed.
042     * @return <code>true</code> if removal is allowed, otherwise <code>false</code>
043     * @throws RemovalNotAllowedException When it is not allowed to remove the entity, because depending items exist
044     */
045    boolean preRemove(T entity) throws RemovalNotAllowedException;
046}

022package org.openwms.core.uaa;
023
024import javax.persistence.DiscriminatorValue;
025import javax.persistence.Entity;
026import java.io.Serializable;
027
028/**
029 * A Grant gives permission to access some kind of application object. Grants to security aware application objects can be permitted or
030 * denied for a certain {@code Role}, depending on the security configuration. Usually {@code Grant}s are assigned to a
031 * {@code Role} and on or more {@code User} s are assigned to each {@code Role}s. A Grant is security aware, that means it is
032 * an concrete {@code SecurityObject}.
033 * <p>
034 * Permissions to UI actions are managed with {@code Grant}s.
035 * </p>
036 * 
037 * @GlossaryTerm
038 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
039 * @version 0.2
040 * @since 0.1
041 * @see User
042 * @see Role
043 * @see SecurityObject
044 */
045@Entity
046@DiscriminatorValue("GRANT")
047public class Grant extends SecurityObject implements Serializable {
048
049    /**
050     * Dear JPA...
051     */
052    protected Grant() {
053    }
054
055    /**
056     * Create a new Grant.
057     * 
058     * @param name
059     *            The name of the {@code Grant}
060     * @param description
061     *            The description text of the {@code Grant}
062     */
063    public Grant(String name, String description) {
064        super(name, description);
065    }
066
067    /**
068     * Create a new Grant.
069     * 
070     * @param name
071     *            The name of the {@code Grant}
072     */
073    public Grant(String name) {
074        super(name);
075    }
076
077    /**
078     * {@inheritDoc}
079     * 
080     * Use the hashCode of the superclass with the hashCode of 'GRANT' to distinguish between {@code Grant}s and other
081     * {@code SecurityObject}s like {@code Role}s.
082     * 
083     * @see SecurityObject#hashCode()
084     */
085    @Override
086    public int hashCode() {
087        final int prime = 31;
088        int result = 1;
089        result = super.hashCode();
090        result = prime * result + "GRANT".hashCode();
091        return result;
092    }
093
094    /**
095     * {@inheritDoc}
096     * 
097     * @see SecurityObject#equals(java.lang.Object)
098     */
099    @Override
100    public boolean equals(Object obj) {
101        if (this == obj) {
102            return true;
103        }
104        if (!(obj instanceof Grant)) {
105            return false;
106        }
107        Grant other = (Grant) obj;
108        if (this.getName() == null) {
109            if (other.getName() != null) {
110                return false;
111            }
112        } else if (!this.getName().equals(other.getName())) {
113            return false;
114        }
115        return true;
116    }
117}

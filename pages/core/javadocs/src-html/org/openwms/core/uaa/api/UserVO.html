022package org.openwms.core.uaa.api;
023
024import javax.validation.constraints.NotNull;
025import javax.validation.constraints.Size;
026import java.io.Serializable;
027import java.util.Date;
028
029import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
030import com.fasterxml.jackson.annotation.JsonProperty;
031import org.ameba.http.AbstractBase;
032
033
034/**
035 * A UserDTO.
036 *
037 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
038 * @version $Revision: $
039 * @since 0.1
040 */
041@JsonIgnoreProperties(ignoreUnknown = true)
042public class UserVO extends AbstractBase implements Serializable {
043
044    private Long id;
045    @JsonProperty("externalUser")
046    private boolean extern;
047    @NotNull
048    @Size(min = 1)
049    private String username;
050    private Date lastPasswordChange;
051    private boolean locked;
052    private String password = "";
053    private boolean enabled;
054    private Date expirationDate;
055    private String fullname = "";
056    private UserDetailsVO userDetails = new UserDetailsVO();
057    private String token;
058    private long version;
059
060    /**
061     * Get the token.
062     *
063     * @return the token.
064     */
065    public String getToken() {
066        return token;
067    }
068
069    /**
070     * Set the token.
071     *
072     * @param token The token to set.
073     */
074    public void setToken(String token) {
075        this.token = token;
076    }
077
078    /**
079     * Get the username.
080     *
081     * @return the username.
082     */
083    public String getUsername() {
084        return username;
085    }
086
087    /**
088     * Set the username.
089     *
090     * @param username The username to set.
091     */
092    public void setUsername(String username) {
093        this.username = username;
094    }
095
096    /**
097     * Get the extern.
098     *
099     * @return the extern.
100     */
101    public boolean isExtern() {
102        return extern;
103    }
104
105    /**
106     * Set the extern.
107     *
108     * @param extern The extern to set.
109     */
110    public void setExtern(boolean extern) {
111        this.extern = extern;
112    }
113
114    /**
115     * Get the lastPasswordChange.
116     *
117     * @return the lastPasswordChange.
118     */
119    public Date getLastPasswordChange() {
120        return lastPasswordChange;
121    }
122
123    /**
124     * Set the lastPasswordChange.
125     *
126     * @param lastPasswordChange The lastPasswordChange to set.
127     */
128    public void setLastPasswordChange(Date lastPasswordChange) {
129        this.lastPasswordChange = lastPasswordChange;
130    }
131
132    /**
133     * Get the locked.
134     *
135     * @return the locked.
136     */
137    public boolean isLocked() {
138        return locked;
139    }
140
141    /**
142     * Set the locked.
143     *
144     * @param locked The locked to set.
145     */
146    public void setLocked(boolean locked) {
147        this.locked = locked;
148    }
149
150    /**
151     * Get the password.
152     *
153     * @return the password.
154     */
155    public String getPassword() {
156        return password;
157    }
158
159    /**
160     * Set the password.
161     *
162     * @param password The password to set.
163     */
164    public void setPassword(String password) {
165        this.password = password;
166    }
167
168    /**
169     * Get the enabled.
170     *
171     * @return the enabled.
172     */
173    public boolean isEnabled() {
174        return enabled;
175    }
176
177    /**
178     * Set the enabled.
179     *
180     * @param enabled The enabled to set.
181     */
182    public void setEnabled(boolean enabled) {
183        this.enabled = enabled;
184    }
185
186    /**
187     * Get the expirationDate.
188     *
189     * @return the expirationDate.
190     */
191    public Date getExpirationDate() {
192        return expirationDate;
193    }
194
195    /**
196     * Set the expirationDate.
197     *
198     * @param expirationDate The expirationDate to set.
199     */
200    public void setExpirationDate(Date expirationDate) {
201        this.expirationDate = expirationDate;
202    }
203
204    /**
205     * Get the fullname.
206     *
207     * @return the fullname.
208     */
209    public String getFullname() {
210        return fullname;
211    }
212
213    /**
214     * Set the fullname.
215     *
216     * @param fullname The fullname to set.
217     */
218    public void setFullname(String fullname) {
219        this.fullname = fullname;
220    }
221
222    /**
223     * Get the userDetails.
224     *
225     * @return the userDetails.
226     */
227    public UserDetailsVO getUserDetails() {
228        return userDetails;
229    }
230
231    /**
232     * Set the userDetails.
233     *
234     * @param userDetails The userDetails to set.
235     */
236    public void setUserDetails(UserDetailsVO userDetails) {
237        this.userDetails = userDetails;
238    }
239
240    /**
241     * Get the id.
242     *
243     * @return the id.
244     */
245    public Long getIdentifer() {
246        return id;
247    }
248
249    /**
250     * Set the id.
251     *
252     * @param id The id to set.
253     */
254    public void setId(Long id) {
255        this.id = id;
256    }
257
258    /**
259     * Get the version.
260     *
261     * @return the version.
262     */
263    public long getVersion() {
264        return version;
265    }
266
267    /**
268     * Set the version.
269     *
270     * @param version The version to set.
271     */
272    public void setVersion(long version) {
273        this.version = version;
274    }
275}

022package org.openwms.core.uaa;
023
024import javax.persistence.Column;
025import javax.persistence.Entity;
026import javax.persistence.ManyToOne;
027import javax.persistence.OrderBy;
028import javax.persistence.Table;
029import java.io.Serializable;
030import java.util.Date;
031
032import org.ameba.integration.jpa.BaseEntity;
033import org.openwms.core.AbstractEntity;
034import org.springframework.util.Assert;
035
036/**
037 * Is a representation of an {@link User} together with her password. <p> When an {@link User} changes her password, the current password is
038 * added to a history list of passwords. This is necessary to omit {@link User}s from setting formerly used passwords. </p>
039 *
040 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
041 * @version 0.2
042 * @GlossaryTerm
043 * @see org.openwms.core.uaa.User
044 * @since 0.1
045 */
046@Entity
047@Table(name = "COR_USER_PASSWORD")
048public class UserPassword extends BaseEntity implements Serializable {
049
050    /** {@link User} assigned to this password. */
051    @ManyToOne
052    private User user;
053    /** Password. */
054    @Column(name = "C_PASSWORD")
055    private String password;
056    /** Date of the last password change. */
057    @Column(name = "C_PASSWORD_CHANGED")
058    @OrderBy
059    private Date passwordChanged = new Date();
060
061    /* ----------------------------- methods ------------------- */
062
063    /**
064     * Create a new {@link UserPassword}.
065     *
066     * @param user The {@link User} to assign
067     * @param password The {@code password} as String to assign
068     * @throws IllegalArgumentException when {@link User} or {@code password} is {@literal null} or empty
069     */
070    public UserPassword(User user, String password) {
071        Assert.notNull(user, "User must not be null");
072        Assert.hasText(password, "Password must not be null");
073        this.user = user;
074        this.password = password;
075    }
076
077    /**
078     * Dear JPA...
079     */
080    protected UserPassword() {
081        super();
082    }
083
084    /**
085     * Return the {@link User} of this password.
086     *
087     * @return The {@link User} of this password
088     */
089    public User getUser() {
090        return user;
091    }
092
093    /**
094     * Change the {@link User}.
095     *
096     * @param user The new {@link User}
097     */
098    public void setUser(User user) {
099        this.user = user;
100    }
101
102    /**
103     * Return the current password.
104     *
105     * @return The current password
106     */
107    public String getPassword() {
108        return password;
109    }
110
111    /**
112     * Return the date of the last password change.
113     *
114     * @return The date when the password has changed
115     */
116    public Date getPasswordChanged() {
117        return new Date(passwordChanged.getTime());
118    }
119
120    /**
121     * {@inheritDoc}
122     * <p>
123     * Does not call the superclass. Uses the password and user for calculation.
124     */
125    @Override
126    public int hashCode() {
127        final int prime = 31;
128        int result = 1;
129        result = prime * result + ((password == null) ? 0 : password.hashCode());
130        result = prime * result + ((user == null) ? 0 : user.hashCode());
131        return result;
132    }
133
134    /**
135     * {@inheritDoc}
136     * <p>
137     * Comparison is done with the business-key (user and password). {@link AbstractEntity#equals(Object)} is not called to avoid comparison
138     * with the UUID.
139     */
140    @Override
141    public boolean equals(Object obj) {
142        if (this == obj) {
143            return true;
144        }
145        if (!(obj instanceof UserPassword)) {
146            return false;
147        }
148        UserPassword other = (UserPassword) obj;
149        if (password == null) {
150            if (other.password != null) {
151                return false;
152            }
153        } else if (!password.equals(other.password)) {
154            return false;
155        }
156        if (user == null) {
157            if (other.user != null) {
158                return false;
159            }
160        } else if (!user.equals(other.user)) {
161            return false;
162        }
163        return true;
164    }
165}

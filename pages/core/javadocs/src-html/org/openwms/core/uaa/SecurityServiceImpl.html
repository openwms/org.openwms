022package org.openwms.core.uaa;
023
024import java.util.ArrayList;
025import java.util.List;
026
027import org.ameba.annotation.TxService;
028import org.ameba.i18n.Translator;
029import org.openwms.core.annotation.FireAfterTransaction;
030import org.openwms.core.event.UserChangedEvent;
031import org.openwms.core.exception.ExceptionCodes;
032import org.slf4j.Logger;
033import org.slf4j.LoggerFactory;
034import org.springframework.beans.factory.annotation.Autowired;
035import org.springframework.util.Assert;
036
037/**
038 * A SecurityServiceImpl is a transactional Spring Service implementation.
039 *
040 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
041 * @version 0.2
042 * @since 0.1
043 */
044@TxService
045public class SecurityServiceImpl implements SecurityService {
046
047    private static final Logger LOGGER = LoggerFactory.getLogger(SecurityServiceImpl.class);
048    @Autowired
049    private SecurityObjectRepository securityObjectRepository;
050    @Autowired
051    private RoleRepository roleRepository;
052    @Autowired
053    private Translator translator;
054
055    /**
056     * {@inheritDoc}
057     * <p>
058     * Triggers {@code UserChangedEvent} after completion.
059     */
060    @Override
061    @FireAfterTransaction(events = {UserChangedEvent.class})
062    public List<Grant> mergeGrants(String moduleName, List<Grant> grants) {
063        Assert.notNull(moduleName, translator.translate(ExceptionCodes.MODULENAME_NOT_NULL));
064        if (LOGGER.isDebugEnabled()) {
065            LOGGER.debug("Merging grants of module:" + moduleName);
066        }
067        List<Grant> persisted = securityObjectRepository.findAllOfModule(moduleName + "%");
068        List<Grant> result = new ArrayList<>(persisted);
069        grants.forEach(g -> {
070            if (persisted.contains(g)) persisted.remove(g);
071            else result.add(securityObjectRepository.save(g));
072        });
073        result.removeAll(persisted);
074        if (!persisted.isEmpty()) {
075            securityObjectRepository.delete(persisted);
076        }
077        return result;
078    }
079}

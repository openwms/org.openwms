058@RestController
059@RequestMapping("/roles")
060public class RolesController extends AbstractWebController {
061
062    @Autowired
063    private RoleService service;
064    @Autowired
065    private BeanMapper m;
066
067    /**
068     * Documented here: https://openwms.atlassian.net/wiki/x/EYAWAQ
069     *
070     * @return JSON response
071     * @status Reviewed [scherrer]
072     */
073    @RequestMapping(method = RequestMethod.GET)
074    @ResponseBody
075    public ResponseEntity<Response<RoleVO>> findAllRoles() {
076        List<RoleVO> roles = m.map(new ArrayList<>(service.findAll()), RoleVO.class);
077        return buildResponse(HttpStatus.OK, translate(Messages.SERVER_OK), Messages.SERVER_OK, roles.toArray(new RoleVO[roles.size()]));
078    }
079
080    /**
081     * Documented here: https://openwms.atlassian.net/wiki/x/BIAWAQ
082     *
083     * @param role The {@link Role} instance to be created
084     * @return An {@link Response} object to encapsulate the result of the creation operation
085     * @status Reviewed [scherrer]
086     */
087    @RequestMapping(method = RequestMethod.POST)
088    @ResponseBody
089    public ResponseEntity<Response<RoleVO>> create(@RequestBody @Valid @NotNull RoleVO role, HttpServletRequest req, HttpServletResponse resp) {
090        RoleVO createdRole = m.map(service.save(m.map(role, Role.class)), RoleVO.class);
091        resp.addHeader(HttpHeaders.LOCATION, getLocationForCreatedResource(req, createdRole.getId().toString()));
092        return buildResponse(HttpStatus.CREATED, translate(Messages.CREATED), Messages.CREATED);
093    }

129    /**
130     * FIXME [scherrer] Comment this
131     *
132     * @param role
133     * @return
134     */
135    @RequestMapping(method = RequestMethod.PUT)
136    @ResponseBody
137    @ResponseStatus(HttpStatus.OK)
138    public RoleVO save(@RequestBody @Valid RoleVO role) {
139        if (role.getId() == null) {
140            String msg = translate(ExceptionCodes.ROLE_IS_TRANSIENT, role.getName());
141            throw new HttpBusinessException(msg, HttpStatus.NOT_ACCEPTABLE);
142        }
143        Role toSave = m.map(role, Role.class);
144        return m.map(service.save(toSave), RoleVO.class);
145    }

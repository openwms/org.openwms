022package org.openwms.core.util;
023
024import java.io.Serializable;
025import java.util.Iterator;
026import java.util.LinkedHashMap;
027import java.util.Map;
028
029/**
030 * A TreeNodeImpl is a simple implementation of a {@link TreeNode}.
031 * 
032 * @param <T>
033 *            The type of the node
034 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
035 * @version $Revision$
036 * @since 0.1
037 */
038public class TreeNodeImpl<T> implements TreeNode<T>, Serializable {
039
040    private static final long serialVersionUID = 5032295350189152303L;
041
042    private T data;
043    private TreeNode<T> parent;
044    private Map<Object, TreeNode<T>> childrenMap = new LinkedHashMap<Object, TreeNode<T>>();
045
046    /**
047     * {@inheritDoc}
048     */
049    @Override
050    public T getData() {
051        return data;
052    }
053
054    /**
055     * {@inheritDoc}
056     */
057    @Override
058    public TreeNode<T> getChild(Object identifier) {
059        return (TreeNode<T>) childrenMap.get(identifier);
060    }
061
062    /**
063     * {@inheritDoc}
064     */
065    @Override
066    public void addChild(Object identifier, TreeNode<T> child) {
067        child.setParent(this);
068        childrenMap.put(identifier, child);
069    }
070
071    /**
072     * {@inheritDoc}
073     */
074    @Override
075    public void removeChild(Object identifier) {
076        TreeNode<T> treeNode = childrenMap.remove(identifier);
077        if (treeNode != null) {
078            treeNode.setParent(null);
079        }
080    }
081
082    /**
083     * {@inheritDoc}
084     */
085    @Override
086    public void setData(T data) {
087        this.data = data;
088    }
089
090    /**
091     * {@inheritDoc}
092     */
093    @Override
094    public TreeNode<T> getParent() {
095        return parent;
096    }
097
098    /**
099     * {@inheritDoc}
100     */
101    @Override
102    public void setParent(TreeNode<T> parent) {
103        this.parent = parent;
104    }
105
106    /**
107     * {@inheritDoc}
108     */
109    @Override
110    public Iterator<Map.Entry<Object, TreeNode<T>>> getChildren() {
111        return childrenMap.entrySet().iterator();
112    }
113
114    /**
115     * {@inheritDoc}
116     */
117    @Override
118    public boolean isLeaf() {
119        return childrenMap.isEmpty();
120    }
121}

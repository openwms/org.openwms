016package org.openwms.core;
017
018import org.ameba.annotation.EnableAspects;
019import org.ameba.app.SolutionApp;
020import org.ameba.i18n.AbstractTranslator;
021import org.ameba.i18n.Translator;
022import org.ameba.mapping.BeanMapper;
023import org.ameba.mapping.DozerMapperImpl;
024import org.springframework.boot.SpringApplication;
025import org.springframework.boot.autoconfigure.SpringBootApplication;
026import org.springframework.context.MessageSource;
027import org.springframework.context.annotation.Bean;
028import org.springframework.context.support.ResourceBundleMessageSource;
029import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
030import org.springframework.security.crypto.password.PasswordEncoder;
031import org.springframework.validation.beanvalidation.LocalValidatorFactoryBean;
032import org.springframework.validation.beanvalidation.MethodValidationPostProcessor;
033
034/**
035 * An UAAStarter.
036 *
037 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
038 */
039@SpringBootApplication(scanBasePackageClasses = {UAAStarter.class, SolutionApp.class})
040@EnableAspects(propagateRootCause = true)
041public class UAAStarter {
042
043    public
044    @Bean
045    PasswordEncoder passwordEncoder() {
046        return new BCryptPasswordEncoder(15);
047    }
048
049    public
050    @Bean
051    Translator translator() {
052        return new AbstractTranslator() {
053            @Override
054            protected MessageSource getMessageSource() {
055                return messageSource();
056            }
057        };
058    }
059
060    public
061    @Bean
062    MessageSource messageSource() {
063        ResourceBundleMessageSource nrrbm = new ResourceBundleMessageSource();
064        nrrbm.setBasename("i18n");
065        return nrrbm;
066    }
067
068    public
069    @Bean
070    BeanMapper beanMapper() {
071        return new DozerMapperImpl("META-INF/dozer/catalog-bean-mappings.xml");
072    }
073
074    public
075    @Bean
076    MethodValidationPostProcessor methodValidationPostProcessor(LocalValidatorFactoryBean validatorFactoryBean) {
077        MethodValidationPostProcessor mvpp = new MethodValidationPostProcessor();
078        mvpp.setValidator(validatorFactoryBean);
079        return mvpp;
080    }
081
082    /**
083     * Boot up!
084     *
085     * @param args Some args
086     */
087    public static void main(String[] args) {
088        SpringApplication.run(UAAStarter.class, args);
089    }
090}

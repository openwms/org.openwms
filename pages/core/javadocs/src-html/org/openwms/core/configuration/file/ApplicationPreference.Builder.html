022package org.openwms.core.configuration.file;
023
024import javax.persistence.Column;
025import javax.persistence.Entity;
026import javax.persistence.EnumType;
027import javax.persistence.Enumerated;
028import javax.persistence.NamedQueries;
029import javax.persistence.NamedQuery;
030import javax.persistence.Table;
031import javax.persistence.UniqueConstraint;
032import javax.xml.bind.annotation.XmlAttribute;
033import javax.xml.bind.annotation.XmlTransient;
034import javax.xml.bind.annotation.XmlType;
035import java.io.Serializable;
036
037import com.fasterxml.jackson.annotation.JsonIgnore;
038import org.openwms.core.configuration.PropertyScope;
039import org.springframework.util.Assert;
040
041/**
042 * An ApplicationPreference is used to store a configuration setting in application scope. <p> The table model of an ApplicationPreference
043 * spans an unique key over the columns C_TYPE and C_KEY. </p> <p> It's counterpart in the context of JAXB is the applicationPreference
044 * element. </p>
045 *
046 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
047 * @version 0.2
048 * @GlossaryTerm
049 * @since 0.1
050 */
051@XmlType(name = "applicationPreference", namespace = "http://www.openwms.org/schema/preferences")
052@Entity
053@Table(name = "COR_APP_PREFERENCE", uniqueConstraints = @UniqueConstraint(columnNames = {"C_TYPE", "C_KEY"}))
054@NamedQueries({@NamedQuery(name = ApplicationPreference.NQ_FIND_BY_OWNER, query = "select ap from ApplicationPreference ap")})
055public class ApplicationPreference extends AbstractPreference implements Serializable {
056
057    /** Query to find all {@code ApplicationPreference}s. Name is {@value}. */
058    public static final String NQ_FIND_BY_OWNER = "ApplicationPreference" + FIND_BY_OWNER;
059
060    /** Type of this preference. */
061    @XmlTransient
062    @Enumerated(EnumType.STRING)
063    @Column(name = "C_TYPE")
064    private PropertyScope type = PropertyScope.APPLICATION;
065
066    /** Key of the preference (not nullable). */
067    @XmlAttribute(name = "key", required = true)
068    @Column(name = "C_KEY", nullable = false)
069    private String key;
070
071    /** Create a new {@code ApplicationPreference}. Only defined by the JAXB implementation. */
072    public ApplicationPreference() {
073        super();
074    }
075
076    /**
077     * Create a new {@code ApplicationPreference}.
078     *
079     * @param key the key
080     * @throws IllegalArgumentException when key is {@literal null} or empty
081     */
082    public ApplicationPreference(String key) {
083        // Called from the client-side only.
084        super();
085        Assert.hasText(key, "Not allowed to create an ApplicationPreference with an empty key");
086        this.key = key;
087    }
088
089    private ApplicationPreference(Builder builder) {
090        setValue(builder.value);
091        binValue = builder.binValue;
092        floatValue = builder.floatValue;
093        setDescription(builder.description);
094        minimum = builder.minimum;
095        maximum = builder.maximum;
096        type = builder.type;
097        key = builder.key;
098        type = PropertyScope.APPLICATION;
099    }
100
101    /**
102     * Get the key.
103     *
104     * @return the key
105     */
106    public String getKey() {
107        return key;
108    }
109
110    /**
111     * {@inheritDoc}
112     */
113    @Override
114    public PropertyScope getType() {
115        return type;
116    }
117
118    /**
119     * {@inheritDoc}
120     */
121    @Override
122    protected Object[] getFields() {
123        return new Object[]{getType(), getKey()};
124    }
125
126    /**
127     * {@inheritDoc}
128     * <p>
129     * Uses the type and key to create a {@link PreferenceKey} instance.
130     *
131     * @see AbstractPreference#getPrefKey()
132     */
133    @Override
134    @JsonIgnore
135    public PreferenceKey getPrefKey() {
136        return new PreferenceKey(getType(), getKey());
137    }
138
139    /**
140     * {@inheritDoc}
141     * <p>
142     * Uses the type and the key for the hashCode calculation.
143     */
144    @Override
145    public int hashCode() {
146        final int prime = 31;
147        int result = super.hashCode();
148        result = prime * result + ((key == null) ? 0 : key.hashCode());
149        result = prime * result + ((type == null) ? 0 : type.hashCode());
150        return result;
151    }
152
153    /**
154     * {@inheritDoc}
155     * <p>
156     * Comparison done with the type and the key fields. Not delegated to super class.
157     */
158    @Override
159    public boolean equals(Object obj) {
160        if (null == obj) {
161            return false;
162        }
163        if (this == obj) {
164            return true;
165        }
166        if (getClass() != obj.getClass()) {
167            return false;
168        }
169        ApplicationPreference other = (ApplicationPreference) obj;
170        if (key == null) {
171            if (other.key != null) {
172                return false;
173            }
174        } else if (!key.equals(other.key)) {
175            return false;
176        }
177        return type == other.type;
178    }
179
180    @Override
181    public String toString() {
182        return "ApplicationPreference{" +
183                "type=" + type +
184                ", key='" + key + '\'' +
185                "} " + super.toString();
186    }
187
188    /**
189     * {@code ApplicationPreference} builder static inner class.
190     */
191    public static final class Builder {
192
193        private PropertyScope type;
194        private String key;
195        private String value;
196        private Serializable binValue;
197        private Float floatValue;
198        private String description;
199        private int minimum;
200        private int maximum;
201
202        public Builder() {
203        }
204
205        /**
206         * Sets the {@code key} and returns a reference to this Builder so that the methods can be chained together.
207         *
208         * @param val the {@code key} to set
209         * @return a reference to this Builder
210         */
211        public Builder withKey(String val) {
212            key = val;
213            return this;
214        }
215
216        /**
217         * Returns a {@code ApplicationPreference} built from the parameters previously set.
218         *
219         * @return a {@code ApplicationPreference} built with parameters of this {@code ApplicationPreference.Builder}
220         */
221        public ApplicationPreference build() {
222            return new ApplicationPreference(this);
223        }
224
225        /**
226         * Sets the {@code value} and returns a reference to this Builder so that the methods can be chained together.
227         *
228         * @param val the {@code value} to set
229         * @return a reference to this Builder
230         */
231        public Builder withValue(String val) {
232            value = val;
233            return this;
234        }
235
236        /**
237         * Sets the {@code binValue} and returns a reference to this Builder so that the methods can be chained together.
238         *
239         * @param val the {@code binValue} to set
240         * @return a reference to this Builder
241         */
242        public Builder withBinValue(Serializable val) {
243            binValue = val;
244            return this;
245        }
246
247        /**
248         * Sets the {@code floatValue} and returns a reference to this Builder so that the methods can be chained together.
249         *
250         * @param val the {@code floatValue} to set
251         * @return a reference to this Builder
252         */
253        public Builder withFloatValue(Float val) {
254            floatValue = val;
255            return this;
256        }
257
258        /**
259         * Sets the {@code description} and returns a reference to this Builder so that the methods can be chained together.
260         *
261         * @param val the {@code description} to set
262         * @return a reference to this Builder
263         */
264        public Builder withDescription(String val) {
265            description = val;
266            return this;
267        }
268
269        /**
270         * Sets the {@code minimum} and returns a reference to this Builder so that the methods can be chained together.
271         *
272         * @param val the {@code minimum} to set
273         * @return a reference to this Builder
274         */
275        public Builder withMinimum(int val) {
276            minimum = val;
277            return this;
278        }
279
280        /**
281         * Sets the {@code maximum} and returns a reference to this Builder so that the methods can be chained together.
282         *
283         * @param val the {@code maximum} to set
284         * @return a reference to this Builder
285         */
286        public Builder withMaximum(int val) {
287            maximum = val;
288            return this;
289        }
290    }
291}

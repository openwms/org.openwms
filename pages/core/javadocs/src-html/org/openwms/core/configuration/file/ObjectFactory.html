022package org.openwms.core.configuration.file;
023
024import javax.xml.bind.annotation.XmlRegistry;
025
026/**
027 * An ObjectFactory allows you to programatically construct new instances of the Java representation for XML content. The Java
028 * representation of XML content can consist of schema derived interfaces and classes representing the binding of schema type definitions,
029 * element declarations and model groups. Factory methods for each of these are provided in this class.
030 * 
031 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
032 * @version 0.2
033 * @since 0.1
034 */
035@XmlRegistry
036public class ObjectFactory {
037
038    /**
039     * Create a new ObjectFactory that can be used to create new instances of schema derived classes for package:
040     * org.openwms.core.domain.preferences.
041     */
042    public ObjectFactory() {
043        super();
044    }
045
046    /**
047     * Create an instance of {@link ModulePreference}.
048     * 
049     * @return an instance of {@link ModulePreference}
050     */
051    public ModulePreference createModulePreference() {
052        return new ModulePreference();
053    }
054
055    /**
056     * Create an instance of {@link Preferences}.
057     * 
058     * @return an instance of {@link Preferences}
059     */
060    public Preferences createPreferences() {
061        return new Preferences();
062    }
063
064    /**
065     * Create an instance of {@link ApplicationPreference}.
066     * 
067     * @return an instance of {@link ApplicationPreference}
068     */
069    public ApplicationPreference createApplicationPreference() {
070        return new ApplicationPreference();
071    }
072
073    /**
074     * Create an instance of {@link UserPreference}.
075     *
076     * @return an instance of {@link UserPreference}
077     */
078    public UserPreference createUserPreference() {
079        return new UserPreference();
080    }
081
082    /**
083     * Create an instance of {@link UserPreference}.
084     *
085     * @param username
086     *            The name of the User
087     * @param key
088     *            The key of the preference
089     * @param description
090     *            The description text
091     * @return an instance of {@link UserPreference}
092     */
093    public static UserPreference createUserPreference(String username, String key, String description) {
094        UserPreference result = new UserPreference(username, key);
095        result.setDescription(description);
096        return result;
097    }
098
099    /**
100     * Create an instance of {@link RolePreference}.
101     *
102     * @return an instance of {@link RolePreference}
103     */
104    public RolePreference createRolePreference() {
105        return new RolePreference();
106    }
107}

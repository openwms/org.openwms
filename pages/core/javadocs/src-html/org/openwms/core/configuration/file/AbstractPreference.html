022package org.openwms.core.configuration.file;
023
024import javax.persistence.Column;
025import javax.persistence.Entity;
026import javax.persistence.GeneratedValue;
027import javax.persistence.Id;
028import javax.persistence.Inheritance;
029import javax.persistence.InheritanceType;
030import javax.persistence.Lob;
031import javax.persistence.Table;
032import javax.xml.bind.annotation.XmlAttribute;
033import javax.xml.bind.annotation.XmlTransient;
034import javax.xml.bind.annotation.XmlType;
035import javax.xml.bind.annotation.XmlValue;
036import java.io.Serializable;
037import java.util.Objects;
038
039import com.fasterxml.jackson.annotation.JsonIgnore;
040import org.openwms.core.configuration.PropertyScope;
041
042/**
043 * An AbstractPreference is a superclass for all other preference classes within the application. <p> It encapsulates some common behavior
044 * of preference types. </p>
045 *
046 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
047 * @version 0.2
048 * @since 0.1
049 */
050@XmlType(name = "abstractPreference", propOrder = {"description"}, namespace = "http://www.openwms.org/schema/preferences")
051@Entity
052@Inheritance(strategy = InheritanceType.JOINED)
053@Table(name = "COR_PREFERENCE")
054public abstract class AbstractPreference implements Serializable {
055
056    @Id
057    @GeneratedValue
058    private Long id;
059    /** Suffix for the FIND_BY_OWNER named query. Default {@value} */
060    public static final String FIND_BY_OWNER = ".findByOwner";
061
062    /** The String value of the {@code AbstractPreference}. */
063    @Column(name = "C_VALUE")
064    protected String value;
065
066    /** A binary value for this {@link AbstractPreference}. */
067    @XmlTransient
068    @Lob
069    @Column(name = "C_BINVALUE")
070    protected Serializable binValue;
071
072    /** A float value of the {@link AbstractPreference}. */
073    @XmlAttribute(name = "floatValue")
074    @Column(name = "C_FLOAT_VALUE")
075    protected Float floatValue;
076
077    /** Description text of the {@link AbstractPreference}. */
078    @Column(name = "C_DESCRIPTION")
079    protected String description;
080
081    /** Minimum value. */
082    @XmlAttribute(name = "minimum")
083    @Column(name = "C_MINIMUM")
084    protected int minimum = 0;
085
086    /** Maximum value. */
087    @XmlAttribute(name = "maximum")
088    @Column(name = "C_MAXIMUM")
089    protected int maximum = 0;
090
091    /** Flag to remember if the preference was originally imported from a file. */
092    @XmlTransient
093    @Column(name = "C_FROM_FILE")
094    private boolean fromFile = true;
095
096    /* ----------------------------- methods ------------------- */
097
098    /**
099     * Return the <code>value</code> of the {@link AbstractPreference}.
100     *
101     * @return The value of the {@link AbstractPreference}
102     */
103    @XmlAttribute(name = "val")
104    public String getValue() {
105        return value;
106    }
107
108    /**
109     * Set the <code>value</code> of the {@link AbstractPreference}.
110     *
111     * @param value The value to set
112     */
113    public void setValue(String value) {
114        this.value = value;
115    }
116
117    /**
118     * Get the binValue.
119     *
120     * @return the binValue.
121     */
122    @XmlTransient
123    @JsonIgnore
124    public Serializable getBinValue() {
125        return binValue;
126    }
127
128    /**
129     * Get the <code>floatValue</code> of the {@link AbstractPreference}.
130     *
131     * @return The floatValue of the preference
132     */
133    public Float getFloatValue() {
134        return floatValue;
135    }
136
137    /**
138     * Return the <code>description</code> of the {@link AbstractPreference}.
139     *
140     * @return The description as String
141     */
142    @XmlValue
143    public String getDescription() {
144        return description;
145    }
146
147    /**
148     * Set a <code>description</code> for the {@link AbstractPreference}.
149     *
150     * @param description The description to set
151     */
152    public void setDescription(String description) {
153        this.description = description;
154    }
155
156    /**
157     * Return the possible minimum value of the {@link AbstractPreference}.
158     *
159     * @return The possible minimum value
160     */
161    public int getMinimum() {
162        return minimum;
163    }
164
165    /**
166     * Return the possible maximum value of the {@link AbstractPreference}.
167     *
168     * @return The possible maximum value
169     */
170    public int getMaximum() {
171        return maximum;
172    }
173
174    /**
175     * Return all fields as concatenated String.
176     *
177     * @return fields as String
178     */
179    @JsonIgnore
180    public String getPropertiesAsString() {
181        // TODO [openwms]: 17/05/16
182        return null;
183    }
184
185    @Override
186    public String toString() {
187        return "AbstractPreference{" +
188                "value='" + value + '\'' +
189                ", binValue=" + binValue +
190                ", floatValue=" + floatValue +
191                ", description='" + description + '\'' +
192                ", minimum=" + minimum +
193                ", maximum=" + maximum +
194                '}';
195    }
196
197    /**
198     * {@inheritDoc}
199     */
200    @Override
201    public boolean equals(Object o) {
202        if (this == o) return true;
203        if (o == null || getClass() != o.getClass()) return false;
204        AbstractPreference that = (AbstractPreference) o;
205        return minimum == that.minimum &&
206                maximum == that.maximum &&
207                Objects.equals(value, that.value) &&
208                Objects.equals(binValue, that.binValue) &&
209                Objects.equals(floatValue, that.floatValue) &&
210                Objects.equals(description, that.description);
211    }
212
213    /**
214     * {@inheritDoc}
215     */
216    @Override
217    public int hashCode() {
218        return Objects.hash(value, binValue, floatValue, description, minimum, maximum);
219    }
220
221    /**
222     * Return all fields as an array of objects.
223     *
224     * @return fields as array
225     */
226    protected abstract Object[] getFields();
227
228    /**
229     * Return the particular type of the preference.
230     *
231     * @return The type of the preference
232     */
233    public abstract PropertyScope getType();
234
235    /**
236     * Return a {@link PreferenceKey} of this preference.
237     *
238     * @return A {@link PreferenceKey}
239     */
240    public abstract PreferenceKey getPrefKey();
241}

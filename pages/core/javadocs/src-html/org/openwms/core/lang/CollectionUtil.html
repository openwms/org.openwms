022package org.openwms.core.lang;
023
024import java.util.Collections;
025import java.util.LinkedHashMap;
026import java.util.List;
027import java.util.Map;
028
029/**
030 * A CollectionUtil.
031 * 
032 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
033 * @version $Revision: 1409 $
034 * @since 0.1
035 */
036public final class CollectionUtil {
037
038    private CollectionUtil() {}
039
040    /**
041     * Return the first element of list. If list is <code>null</code> of empty, <code>null</code> is returned.
042     * 
043     * @param <T>
044     *            Any type
045     * @param list
046     *            The {@link List}
047     * @return list[0] or <code>null</code> if list == null || list.isEmpty
048     */
049    public static <T> T getFirstOrNull(List<T> list) {
050        if (list == null || list.isEmpty()) {
051            return null;
052        }
053        return list.get(0);
054    }
055
056    /**
057     * Convert a list into a {@link LinkedHashMap} using a {@link ListExtractor} to determine the keys and values for the new Map.
058     * 
059     * @param <K>
060     *            Type of list key
061     * @param <V>
062     *            Type of list value
063     * @param list
064     *            The list to convert
065     * @param extractor
066     *            To determine key and value for each list entry
067     * @return A created {@link LinkedHashMap} or an empty map when one of the arguments is <code>null</code>
068     */
069    public static <K, V> Map<K, V> asHashMap(List<V> list, ListExtractor<K, V> extractor) {
070        if (list == null || extractor == null || list.isEmpty()) {
071            return Collections.<K, V> emptyMap();
072        }
073        Map<K, V> map = new LinkedHashMap<K, V>(list.size());
074        for (V node : list) {
075            map.put(extractor.extractKey(node), extractor.extractValue(node));
076        }
077        return map;
078    }
079}

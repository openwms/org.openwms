022package org.openwms.core.lang;
023
024import javax.persistence.Column;
025import javax.persistence.Embedded;
026import javax.persistence.Entity;
027import javax.persistence.PostLoad;
028import javax.persistence.Table;
029import javax.persistence.Transient;
030import javax.persistence.UniqueConstraint;
031import java.io.Serializable;
032
033import org.ameba.integration.jpa.BaseEntity;
034import org.springframework.util.Assert;
035
036
037/**
038 * An I18n entity stores multiple translations assigned to an unique key.
039 *
040 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
041 * @version 0.2
042 * @since 0.1
043 */
044@Entity
045@Table(name = "COR_I18N", uniqueConstraints = @UniqueConstraint(columnNames = {"C_KEY", "C_MODULE_NAME"}))
046public class I18n extends BaseEntity implements Serializable {
047
048    /** The natural key is used as references in the application (not nullable). */
049    @Column(name = "C_KEY", nullable = false)
050    private String key;
051    /** The name of the owning {@code Module} to which this translation set belongs to. */
052    @Column(name = "C_MODULE_NAME")
053    private String moduleName = "CORE";
054    /** The translation set of this entity. */
055    @Embedded
056    private I18nSet lang;
057    /**
058     * The cKey is a transient field that is constructed after the entity is loaded from the persistent store. Usually this field is
059     * accessed from the client application to have an unique identifier - a combination of the owning {@code moduleName} and the {@code
060     * key}.
061     */
062    @Transient
063    private String cKey;
064
065    /** Dear JPA... */
066    public I18n() {
067    }
068
069    /**
070     * Create a new I18n.
071     *
072     * @param moduleName The name of the {@code Module} where this entity belongs to
073     * @param key The key to access this translation
074     * @param lang A set of languages
075     * @throws IllegalArgumentException when the {@code moduleName} or the {@code key} is {@literal null} or empty
076     */
077    public I18n(String moduleName, String key, I18nSet lang) {
078        super();
079        Assert.hasText(moduleName, "Not allowed to create an I18n instance with an empty moduleName");
080        Assert.hasText(key, "Not allowed to create an I18n instance with an empty key");
081        this.moduleName = moduleName;
082        this.key = key;
083        this.lang = lang;
084    }
085
086    /**
087     * Create a new I18n.
088     *
089     * @param key The key to access this translation
090     * @param lang A set of languages
091     * @throws IllegalArgumentException when the {@code key} is {@literal null} or empty
092     */
093    public I18n(String key, I18nSet lang) {
094        super();
095        Assert.hasText(key, "Not allowed to create an I18n instance with an empty key");
096        this.key = key;
097        this.lang = lang;
098    }
099
100    /**
101     * After loading the entity, combine the {@code moduleName} field and the {@code key} field. Store the concatenated String in a
102     * transient field {@code cKey}.
103     */
104    @PostLoad
105    protected void onLoad() {
106        cKey = moduleName + key;
107    }
108
109    /**
110     * Get the cKey.
111     *
112     * @return the cKey.
113     */
114    public String getCKey() {
115        return cKey;
116    }
117
118    /**
119     * Get the key.
120     *
121     * @return the key.
122     */
123    public String getKey() {
124        return key;
125    }
126
127    /**
128     * Get the moduleName.
129     *
130     * @return the moduleName.
131     */
132    public String getModuleName() {
133        return moduleName;
134    }
135
136    /**
137     * Get the language set.
138     *
139     * @return the language set.
140     */
141    public I18nSet getLang() {
142        return lang;
143    }
144
145    /**
146     * {@inheritDoc}
147     * <p>
148     * Use {@code key} and {@code moduleName} for calculation.
149     *
150     * @see java.lang.Object#hashCode()
151     */
152    @Override
153    public int hashCode() {
154        final int prime = 31;
155        int result = 42;
156        result = prime * result + ((cKey == null) ? 0 : cKey.hashCode());
157        result = prime * result + ((moduleName == null) ? 0 : moduleName.hashCode());
158        return result;
159    }
160
161    /**
162     * {@inheritDoc}
163     * <p>
164     * Use {@code key} and {@code moduleName} for comparison.
165     *
166     * @see java.lang.Object#equals(java.lang.Object)
167     */
168    @Override
169    public boolean equals(Object obj) {
170        if (this == obj) {
171            return true;
172        }
173        if (!(obj instanceof I18n)) {
174            return false;
175        }
176        I18n other = (I18n) obj;
177        if (key == null) {
178            if (other.key != null) {
179                return false;
180            }
181        } else if (!key.equals(other.key)) {
182            return false;
183        }
184        if (moduleName == null) {
185            if (other.moduleName != null) {
186                return false;
187            }
188        } else if (!moduleName.equals(other.moduleName)) {
189            return false;
190        }
191        return true;
192    }
193}

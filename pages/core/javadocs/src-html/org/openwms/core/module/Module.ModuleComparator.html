022package org.openwms.core.module;
023
024import javax.persistence.Column;
025import javax.persistence.Entity;
026import javax.persistence.NamedQueries;
027import javax.persistence.NamedQuery;
028import javax.persistence.OrderBy;
029import javax.persistence.Table;
030import javax.persistence.Transient;
031import java.io.Serializable;
032import java.util.Comparator;
033
034import org.ameba.integration.jpa.BaseEntity;
035import org.openwms.core.AbstractEntity;
036import org.openwms.core.DomainObject;
037import org.openwms.core.values.CoreTypeDefinitions;
038import org.springframework.util.Assert;
039
040/**
041 * A Module represents an Adobe Flex Module and is used to store some basic information about that module, i.e. a name, an URL where the
042 * module from, or whether the Adobe Flex Module should be loaded on application startup.
043 * 
044 * @GlossaryTerm
045 * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
046 * @version $Revision$
047 * @since 0.1
048 * @see AbstractEntity
049 * @see DomainObject
050 */
051@Entity
052@Table(name = "COR_MODULE")
053@NamedQueries({ @NamedQuery(name = Module.NQ_FIND_ALL, query = "select m from Module m order by m.startupOrder"),
054        @NamedQuery(name = Module.NQ_FIND_BY_UNIQUE_QUERY, query = "select m from Module m where m.moduleName = ?1") })
055public class Module extends BaseEntity implements Serializable {
056
057    /**
058     * Unique name of the <code>Module</code> (natural key, unique, not-null).
059     */
060    @Column(name = "C_MODULE_NAME", unique = true, nullable = false)
061    private String moduleName;
062    /**
063     * URL from where to load this <code>Module</code> (unique, not-null).
064     */
065    @Column(name = "C_URL", unique = true, nullable = false)
066    private String url;
067    /**
068     * Flag used on the client side to store whether the <code>Module</code> is actually loaded or not. It's a dynamic value and not
069     * persisted.
070     */
071    @Transient
072    private boolean loaded = false;
073    /**
074     * <code>true</code> when the <code>Module</code> should be loaded on application startup.
075     */
076    @Column(name = "C_LOAD_ON_STARTUP")
077    private boolean loadOnStartup = true;
078    /**
079     * Defines the startup order compared with other Modules. Modules with a lower <code>startupOrder</code> are loaded before this one.
080     */
081    @Column(name = "C_STARTUP_ORDER")
082    @OrderBy
083    private int startupOrder;
084    /**
085     * A description text of this <code>Module</code>.
086     */
087    @Column(name = "C_DESCRIPTION", length = CoreTypeDefinitions.DESCRIPTION_LENGTH)
088    private String description = "--";
089    /**
090     * Query to find all <code>Module</code>s. Name is {@value} .
091     */
092    public static final String NQ_FIND_ALL = "Module.findAll";
093    /**
094     * Query to find <strong>one</strong> <code>Module</code> by its natural key. <li>
095     * Query parameter index <strong>1</strong> : The <code>moduleName</code> of the <code>Module</code> to search for</li><br />
096     * Name is {@value} .
097     */
098    public static final String NQ_FIND_BY_UNIQUE_QUERY = "Module.findByModuleName";
099
100    /**
101     * A ModuleComparator.
102     * 
103     * @author <a href="mailto:scherrer@openwms.org">Heiko Scherrer</a>
104     * @version $Revision$
105     * @since 0.2
106     */
107    public static class ModuleComparator implements Comparator<Module>, Serializable {
108
109        private static final long serialVersionUID = 8749015473190257293L;
110
111        /**
112         * {@inheritDoc}
113         * 
114         * Return 1 when the startupOrder of o1 is greater or equals than the startupOrder of o2, -1 when it is less.
115         * 
116         * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
117         */
118        @Override
119        public int compare(Module o1, Module o2) {
120            return o1.getStartupOrder() >= o2.getStartupOrder() ? 1 : -1;
121        }
122    };
123
124    /**
125     * Dear JPA ...
126     */
127    protected Module() {
128    }
129
130    /**
131     * Create a new <code>Module</code>.
132     * 
133     * @param moduleName
134     *            The unique <code>Module</code> name
135     * @param url
136     *            The unique URL
137     * @throws IllegalArgumentException
138     *             in case the new moduleName is <code>null</code> or empty
139     */
140    public Module(String moduleName, String url) {
141        super();
142        Assert.hasText(moduleName, "Not allowed to set the moduleName to null or an empty String");
143        this.moduleName = moduleName;
144        this.url = url;
145    }
146
147    /**
148     * Get the <code>moduleName</code>.
149     * 
150     * @return the moduleName.
151     */
152    public String getModuleName() {
153        return moduleName;
154    }
155
156    /**
157     * Set the <code>moduleName</code>.
158     * 
159     * @param moduleName
160     *            The moduleName to set
161     * @throws IllegalArgumentException
162     *             in case the new moduleName is <code>null</code> or empty
163     */
164    public void setModuleName(String moduleName) {
165        Assert.hasText(moduleName, "Not allowed to set the moduleName to null or an empty String");
166        this.moduleName = moduleName;
167    }
168
169    /**
170     * Get the <code>url</code>.
171     * 
172     * @return the url
173     */
174    public String getUrl() {
175        return url;
176    }
177
178    /**
179     * Set the <code>url</code>.
180     * 
181     * @param url
182     *            The url to set
183     * @throws IllegalArgumentException
184     *             in case the new url is <code>null</code> or empty
185     */
186    public void setUrl(String url) {
187        Assert.hasText(url, "Not allowed to set the url to null or an empty String");
188        this.url = url;
189    }
190
191    /**
192     * Is the <code>Module</code> currently loaded.
193     * 
194     * @return <code>true</code> if loaded, otherwise <code>false</code>
195     */
196    public boolean isLoaded() {
197        return loaded;
198    }
199
200    /**
201     * Set the <code>loaded</code> flag.
202     * 
203     * @param loaded
204     *            The loaded to set
205     */
206    public void setLoaded(boolean loaded) {
207        this.loaded = loaded;
208    }
209
210    /**
211     * Get the <code>description</code>.
212     * 
213     * @return the description
214     */
215    public String getDescription() {
216        return description;
217    }
218
219    /**
220     * Set the <code>description</code>.
221     * 
222     * @param description
223     *            The description to set
224     */
225    public void setDescription(String description) {
226        this.description = description;
227    }
228
229    /**
230     * Should the <code>Module</code> be loaded on application startup.
231     * 
232     * @return <code>true</code> if the <code>Module</code> should be loaded on application startup, otherwise <code>false</code>
233     */
234    public boolean isLoadOnStartup() {
235        return loadOnStartup;
236    }
237
238    /**
239     * Set the <code>loadOnStartup</code> flag.
240     * 
241     * @param loadOnStartup
242     *            The loadOnStartup to set
243     */
244    public void setLoadOnStartup(boolean loadOnStartup) {
245        this.loadOnStartup = loadOnStartup;
246    }
247
248    /**
249     * Get the <code>startupOrder</code>.
250     * 
251     * @return the startupOrder
252     */
253    public int getStartupOrder() {
254        return startupOrder;
255    }
256
257    /**
258     * Set the <code>startupOrder</code>.
259     * 
260     * @param startupOrder
261     *            The startupOrder to set
262     */
263    public void setStartupOrder(int startupOrder) {
264        this.startupOrder = startupOrder;
265    }
266
267    /**
268     * Returns the <code>moduleName</code>.
269     * 
270     * @see java.lang.Object#toString()
271     * @return The moduleName
272     */
273    @Override
274    public String toString() {
275        return moduleName;
276    }
277}
